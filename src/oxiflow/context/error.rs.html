1//! # Module `context::error`
2//!
3//! Main error type of the oxiflow engine (DD-004, issue #28).
4//!
5//! `OxiflowError` is a typed enum built with `thiserror` covering all engine
6//! failure cases. Every public function returns `Result<_, OxiflowError>` —
7//! never `Result<_, String>`.
8//!
9//! ## Design rationale
10//!
11//! chrom-rs used `Result<_, String>` throughout. String errors are not
12//! matchable programmatically: downstream code cannot distinguish a missing
13//! calculator from a solver divergence without parsing strings. `OxiflowError`
14//! makes every failure case a first-class type.
15
16use crate::context::variable::ContextVariable;
17
18/// Typed error enum for all oxiflow engine failures.
19///
20/// Each variant corresponds to a distinct, matchable failure mode.
21/// The `source` field in `ComputationFailed` preserves the original error
22/// for error-chain display while keeping the variant matchable.
23///
24/// # Serialisation
25///
26/// `OxiflowError` does not implement `serde::Serialize` / `serde::Deserialize`.
27/// The `ComputationFailed` variant holds a `Box<dyn std::error::Error + Send + Sync>`
28/// (a trait object), which cannot be serialised directly. Error context for
29/// post-mortem analysis is captured as a `String` in `SimulationSnapshot`
30/// (DD-025 Option B, v0.6.0).
31///
32/// # Examples
33///
34/// ```rust
35/// use oxiflow::context::error::OxiflowError;
36/// use oxiflow::context::variable::ContextVariable;
37///
38/// let err = OxiflowError::MissingCalculator(ContextVariable::Time);
39/// assert!(matches!(err, OxiflowError::MissingCalculator(ContextVariable::Time)));
40///
41/// let err = OxiflowError::TypeMismatch {
42///     expected: "Scalar",
43///     actual:   "Vector",
44/// };
45/// assert!(matches!(err, OxiflowError::TypeMismatch { .. }));
46/// ```
47#[derive(Debug, thiserror::Error)]
48pub enum OxiflowError {
49    /// No calculator registered for the required variable.
50    #[error("missing calculator for variable: {0}")]
51    MissingCalculator(ContextVariable),
52
53    /// A calculator returned an error while computing a variable.
54    #[error("computation failed for {variable}: {source}")]
55    ComputationFailed {
56        variable: ContextVariable,
57        #[source]
58        source: Box<dyn std::error::Error + Send + Sync>,
59    },
60
61    /// A circular dependency was detected among calculators.
62    #[error("circular dependency detected involving: {0}")]
63    CircularDependency(ContextVariable),
64
65    /// A context accessor was called with the wrong `ContextValue` variant.
66    #[error("type mismatch: expected {expected}, actual {actual}")]
67    TypeMismatch {
68        expected: &'static str,
69        actual: &'static str,
70    },
71
72    /// The mesh or domain configuration is invalid.
73    #[error("invalid domain: {0}")]
74    InvalidDomain(String),
75
76    /// A required physical or numerical precondition was violated.
77    ///
78    /// Used when a computation cannot proceed due to an invalid parameter value
79    /// or an inconsistent input state. Distinct from [`ComputationFailed`] which
80    /// wraps an upstream error: `PreconditionFailed` is raised by the component
81    /// itself upon detecting a violated invariant.
82    ///
83    /// `context` identifies the component that raised the error (e.g.
84    /// `"DanckwertsInlet"`, `"UniformGrid1D"`). `message` describes the
85    /// violated condition and may include dynamic values.
86    ///
87    /// [`ComputationFailed`]: OxiflowError::ComputationFailed
88    #[error("precondition failed in {context}: {message}")]
89    PreconditionFailed {
90        /// Name of the component that detected the violation.
91        context: &'static str,
92        /// Human-readable description of the violated condition.
93        message: String,
94    },
95
96    /// An external data source returned an error or is unavailable.
97    #[error("external data error: {0}")]
98    ExternalData(String),
99
100    /// The solver produced a non-finite state and cannot continue.
101    #[error("solver divergence at t={time:.4e}: {reason}")]
102    SolverDivergence { time: f64, reason: String },
103}

111    #[test]
112    fn missing_calculator_matches_variable() {
113        let err = OxiflowError::MissingCalculator(ContextVariable::Time);

126    #[test]
127    fn computation_failed_is_matchable() {
128        let source: Box<dyn std::error::Error + Send + Sync> =
129            Box::new(std::io::Error::other("calculator error"));
130        let err = OxiflowError::ComputationFailed {
131            variable: ContextVariable::SpatialGradient {
132                dimension: 0,
133                component: None,
134            },
135            source,
136        };

138    }
