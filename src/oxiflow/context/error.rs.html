1//! # Module `context::error`
2//!
3//! Main error type of the oxiflow engine (DD-004, issue #28).
4//!
5//! `OxiflowError` is a typed enum built with `thiserror` covering all engine
6//! failure cases. Every public function returns `Result<_, OxiflowError>` —
7//! never `Result<_, String>`.
8//!
9//! ## Design rationale
10//!
11//! chrom-rs used `Result<_, String>` throughout. String errors are not
12//! matchable programmatically: downstream code cannot distinguish a missing
13//! calculator from a solver divergence without parsing strings. `OxiflowError`
14//! makes every failure case a first-class type.
15
16use crate::context::variable::ContextVariable;
17
18/// Typed error enum for all oxiflow engine failures.
19///
20/// Each variant corresponds to a distinct, matchable failure mode.
21/// The `source` field in `ComputationFailed` preserves the original error
22/// for error-chain display while keeping the variant matchable.
23///
24/// # Examples
25///
26/// ```rust
27/// use oxiflow::context::error::OxiflowError;
28/// use oxiflow::context::variable::ContextVariable;
29///
30/// let err = OxiflowError::MissingCalculator(ContextVariable::Time);
31/// assert!(matches!(err, OxiflowError::MissingCalculator(ContextVariable::Time)));
32///
33/// let err = OxiflowError::TypeMismatch {
34///     expected: "Scalar",
35///     actual:   "Vector",
36/// };
37/// assert!(matches!(err, OxiflowError::TypeMismatch { .. }));
38/// ```
39#[derive(Debug, thiserror::Error)]
40pub enum OxiflowError {
41    /// No calculator registered for the required variable.
42    #[error("missing calculator for variable: {0}")]
43    MissingCalculator(ContextVariable),
44
45    /// A calculator returned an error while computing a variable.
46    #[error("computation failed for {variable}: {source}")]
47    ComputationFailed {
48        variable: ContextVariable,
49        #[source]
50        source: Box<dyn std::error::Error + Send + Sync>,
51    },
52
53    /// A circular dependency was detected among calculators.
54    #[error("circular dependency detected involving: {0}")]
55    CircularDependency(ContextVariable),
56
57    /// A context accessor was called with the wrong `ContextValue` variant.
58    #[error("type mismatch: expected {expected}, actual {actual}")]
59    TypeMismatch {
60        expected: &'static str,
61        actual: &'static str,
62    },
63
64    /// The mesh or domain configuration is invalid.
65    #[error("invalid domain: {0}")]
66    InvalidDomain(String),
67
68    /// An external data source returned an error or is unavailable.
69    #[error("external data error: {0}")]
70    ExternalData(String),
71
72    /// The solver produced a non-finite state and cannot continue.
73    #[error("solver divergence at t={time:.4e}: {reason}")]
74    SolverDivergence { time: f64, reason: String },
75}

79#[cfg(test)]
80mod tests {
81    use super::*;
82
83    #[test]
84    fn missing_calculator_matches_variable() {
85        let err = OxiflowError::MissingCalculator(ContextVariable::Time);
86        assert!(matches!(
87            err,
88            OxiflowError::MissingCalculator(ContextVariable::Time)
89        ));
90    }
91
92    #[test]
93    fn missing_calculator_display_contains_variable() {
94        let err = OxiflowError::MissingCalculator(ContextVariable::TimeStep);
95        assert!(err.to_string().contains("TimeStep"));
96    }
97
98    #[test]
99    fn computation_failed_is_matchable() {
100        let source: Box<dyn std::error::Error + Send + Sync> =
101            Box::new(std::io::Error::other("calculator error"));
102        let err = OxiflowError::ComputationFailed {
103            variable: ContextVariable::SpatialGradient {
104                dimension: 0,
105                component: None,
106            },
107            source,
108        };
109        assert!(matches!(err, OxiflowError::ComputationFailed { .. }));
110    }
111
112    #[test]
113    fn computation_failed_display_contains_variable_and_source() {
114        let source: Box<dyn std::error::Error + Send + Sync> =
115            Box::new(std::io::Error::other("overflow"));
116        let err = OxiflowError::ComputationFailed {
117            variable: ContextVariable::SpatialGradient {
118                dimension: 1,
119                component: None,
120            },
121            source,
122        };
123        let msg = err.to_string();
124        assert!(msg.contains("SpatialGradient"));
125        assert!(msg.contains("overflow"));
126    }

128    #[test]
129    fn circular_dependency_matches_variable() {
