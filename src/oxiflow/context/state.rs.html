1//! # Module `context::state`
2//!
3//! Multi-domain state container for coupled simulations (DD-010, DD-011).
4//!
5//! ## Design
6//!
7//! [`MultiDomainState`] maps a composite key `(DomainId, PhysicalQuantity)`
8//! to a [`ContextValue`] field. This is **Option B** from the design review
9//! (2026-06-01): the key is fully typed and self-documenting — no implicit
10//! column-ordering convention is required.
11//!
12//! ```text
13//! ("column", Concentration { component: 0 }) → ScalarField([0.0, 1.2, …])
14//! ("column", Concentration { component: 1 }) → ScalarField([0.0, 0.3, …])
15//! ("lake",   Velocity      { component: 0 }) → ScalarField([0.1, 0.2, …])
16//! ```
17//!
18//! ## Serialisation (feature `serde`)
19//!
20//! `MultiDomainState` serialises to an array of explicit entry objects for
21//! maximum readability by external tools (Python, CLI, wiki):
22//!
23//! ```json
24//! {
25//!   "states": [
26//!     { "domain": "column", "quantity": "Concentration", "component": 0, "value": [0.0, 1.2] },
27//!     { "domain": "lake",   "quantity": "Velocity",      "component": 0, "value": [0.1, 0.2] }
28//!   ]
29//! }
30//! ```
31//!
32//! `Custom` quantities serialise the `name` field directly as `"quantity"` —
33//! no `"Custom(...)"` wrapper. `Temperature` and `Pressure` always serialise
34//! with `"component": 0`.
35//!
36//! ## Memory layout
37//!
38//! Field values are stored as [`ContextValue::ScalarField`] (`DVector<f64>`,
39//! column-major) or [`ContextValue::VectorField`] (`DMatrix<f64>`, column-major
40//! — nalgebra default). This satisfies DD-026 INV-GPU-1 and INV-GPU-4.
41//!

51// ── MultiDomainState ──────────────────────────────────────────────────────────
52
53/// Multi-domain physical state container.
54///
55/// Maps `(DomainId, PhysicalQuantity)` composite keys to [`ContextValue`]
56/// fields. Used by [`CouplingOperator`](crate::coupling) to exchange state
57/// between physical domains without implicit ordering conventions (DD-010,
58/// DD-011, INV-3).
59///
60/// # Examples
61///
62/// ```rust
63/// use oxiflow::context::state::MultiDomainState;
64/// use oxiflow::context::quantity::PhysicalQuantity;
65/// use oxiflow::context::value::ContextValue;
66/// use oxiflow::solver::scenario::DomainId;
67/// use nalgebra::DVector;
68///
69/// let mut state = MultiDomainState::new();
70///
71/// state.set(
72///     DomainId::new("column"),
73///     PhysicalQuantity::concentration(),
74///     ContextValue::ScalarField(DVector::from_vec(vec![0.0, 1.0, 2.0])),
75/// );
76///
77/// let field = state.get(&DomainId::new("column"), &PhysicalQuantity::concentration());
78/// assert!(field.is_some());
79/// ```
80#[derive(Debug, Clone)]
81#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
82#[cfg_attr(feature = "serde", serde(into = "MultiDomainStateSerde"))]
83#[cfg_attr(feature = "serde", serde(try_from = "MultiDomainStateSerde"))]
84pub struct MultiDomainState {
85    /// Internal map: (domain, quantity) → field value.
86    ///
87    /// Column-major memory layout for all field payloads (DD-026 INV-GPU-4).
88    states: HashMap<(DomainId, PhysicalQuantity), ContextValue>,
89}
90
91impl MultiDomainState {
92    // ── Constructors ──────────────────────────────────────────────────────────
93
94    /// Creates an empty state container.
95    pub fn new() -> Self {
96        Self {
97            states: HashMap::new(),
98        }
99    }
100
101    // ── Accessors ─────────────────────────────────────────────────────────────
102
103    /// Returns a reference to the field for `(domain, quantity)`, if present.
104    pub fn get(&self, domain: &DomainId, quantity: &PhysicalQuantity) -> Option<&ContextValue> {
105        self.states.get(&(domain.clone(), quantity.clone()))
106    }
107
108    /// Returns a mutable reference to the field for `(domain, quantity)`, if present.
109    pub fn get_mut(
110        &mut self,
111        domain: &DomainId,
112        quantity: &PhysicalQuantity,
113    ) -> Option<&mut ContextValue> {
114        self.states.get_mut(&(domain.clone(), quantity.clone()))
115    }
116
117    /// Inserts or replaces the field for `(domain, quantity)`.
118    pub fn set(&mut self, domain: DomainId, quantity: PhysicalQuantity, value: ContextValue) {
119        self.states.insert((domain, quantity), value);
120    }
121
122    /// Removes the field for `(domain, quantity)`, returning it if present.
123    pub fn remove(
124        &mut self,
125        domain: &DomainId,
126        quantity: &PhysicalQuantity,
127    ) -> Option<ContextValue> {
128        self.states.remove(&(domain.clone(), quantity.clone()))
129    }
130
131    /// Returns `true` if no fields are stored.
132    pub fn is_empty(&self) -> bool {
133        self.states.is_empty()
134    }
135
136    /// Returns the number of stored fields.
137    pub fn len(&self) -> usize {
138        self.states.len()
139    }
140
141    /// Returns an iterator over all `((domain, quantity), value)` entries.
142    pub fn iter(&self) -> impl Iterator<Item = (&DomainId, &PhysicalQuantity, &ContextValue)> {
143        self.states
144            .iter()
145            .map(|((domain, quantity), value)| (domain, quantity, value))
146    }
147
148    // ── Domain helpers ────────────────────────────────────────────────────────
149
150    /// Returns all fields belonging to `domain`.
151    pub fn domain_fields<'a>(
152        &'a self,
153        domain: &'a DomainId,
154    ) -> impl Iterator<Item = (&'a PhysicalQuantity, &'a ContextValue)> + 'a {
155        self.states
156            .iter()
157            .filter(move |((d, _), _)| d == domain)
158            .map(|((_, q), v)| (q, v))
159    }
160
161    /// Returns `true` if `domain` has at least one field stored.
162    pub fn contains_domain(&self, domain: &DomainId) -> bool {
163        self.states.keys().any(|(d, _)| d == domain)
164    }
165
166    // ── Validation ────────────────────────────────────────────────────────────
167
168    /// Verifies that all fields for a domain have the same number of DOFs.
169    ///
170    /// Returns `OxiflowError::InvalidDomain` if inconsistent field lengths are
171    /// detected for `domain`.
172    pub fn validate_domain_consistency(&self, domain: &DomainId) -> Result<(), OxiflowError> {
173        let lengths: Vec<usize> = self
174            .domain_fields(domain)
175            .filter_map(|(_, v)| match v {
176                ContextValue::ScalarField(f) => Some(f.len()),
177                ContextValue::VectorField(f) => Some(f.nrows()),
178                _ => None,
179            })
180            .collect();
181
182        if lengths.windows(2).any(|w| w[0] != w[1]) {
183            return Err(OxiflowError::InvalidDomain(format!(
184                "domain '{}' has fields with inconsistent DOF counts: {:?}",
185                domain, lengths
186            )));
187        }
188        Ok(())
189    }
190}
191
192impl Default for MultiDomainState {
193    fn default() -> Self {
194        Self::new()
195    }
196}

198// ── Serde support ─────────────────────────────────────────────────────────────
199
200/// Serialisation entry — one per `(domain, quantity)` pair.
201///
202/// Produces the human-readable JSON format consumed by external tools:
203///
204/// ```json
205/// { "domain": "column", "quantity": "Concentration", "component": 0, "value": [...] }
206/// ```
207#[cfg(feature = "serde")]
208#[derive(serde::Serialize, serde::Deserialize)]
209struct MultiDomainStateEntry {
210    domain: String,
211    quantity: String,
212    component: usize,
213    value: ContextValue,
214}
215
216/// Newtype wrapper for serde — holds the flat array of entries.
217#[cfg(feature = "serde")]
218#[derive(serde::Serialize, serde::Deserialize)]
219struct MultiDomainStateSerde {
220    states: Vec<MultiDomainStateEntry>,
221}
222
223#[cfg(feature = "serde")]
224impl From<MultiDomainState> for MultiDomainStateSerde {
225    fn from(mds: MultiDomainState) -> Self {
226        let states = mds
227            .states
228            .into_iter()
229            .map(|((domain, quantity), value)| MultiDomainStateEntry {
230                domain: domain.as_str().to_string(),
231                quantity: quantity.kind_str().to_string(),
232                component: quantity.component(),
233                value,
234            })
235            .collect();
236        Self { states }
237    }
238}
239
240#[cfg(feature = "serde")]
241impl TryFrom<MultiDomainStateSerde> for MultiDomainState {
242    type Error = OxiflowError;
243
244    fn try_from(raw: MultiDomainStateSerde) -> Result<Self, Self::Error> {
245        let mut mds = MultiDomainState::new();
246        for entry in raw.states {
247            let domain = DomainId::new(entry.domain);
248            let quantity = parse_quantity(&entry.quantity, entry.component)?;
249            mds.set(domain, quantity, entry.value);
250        }
251        Ok(mds)
252    }
253}
254
255/// Reconstructs a [`PhysicalQuantity`] from its serialised `kind_str` and
256/// `component` index.
257#[cfg(feature = "serde")]
258fn parse_quantity(kind: &str, component: usize) -> Result<PhysicalQuantity, OxiflowError> {
259    match kind {
260        "Concentration" => Ok(PhysicalQuantity::Concentration { component }),
261        "Temperature" => Ok(PhysicalQuantity::Temperature),
262        "Pressure" => Ok(PhysicalQuantity::Pressure),
263        "Velocity" => Ok(PhysicalQuantity::Velocity { component }),
264        name => Ok(PhysicalQuantity::Custom {
265            name: name.to_string().into(),
266            component,
267        }),
268    }
269}
