1//! # Module `boundary::danckwerts`
2//!
3//! Danckwerts boundary conditions for convection-diffusion problems (issue #35).
4//!
5//! ## Physical context
6//!
7//! Danckwerts conditions arise from a mass balance at the boundaries of a porous
8//! medium column (chromatography, fixed-bed reactors, dispersion in soil). They
9//! account for both convective and dispersive fluxes at the inlet and impose zero
10//! dispersive flux at the outlet.
11//!
12//! ## Inlet — Robin condition
13//!
14//! At $x = 0$, the total mass flux is continuous:
15//!
16//! $$v \, u(0, t) - D \left.\frac{\partial u}{\partial x}\right|_{x=0}
17//!   = v \, u_{\text{feed}}(t)$$
18//!
19//! Rearranged into an explicit assignment for the inlet node:
20//!
21//! $$u(0, t) = u_{\text{feed}}(t) + \frac{D}{v}
22//!   \left.\frac{\partial u}{\partial x}\right|_{x=0}$$
23//!
24//! The feed concentration $u_{\text{feed}}(t)$ is read from [`ComputeContext`] via
25//! a typed [`ContextVariable`] key — it can be constant, time-dependent, or
26//! produced by any user-supplied [`ContextCalculator`].
27//!
28//! ## Outlet — Neumann condition
29//!
30//! At $x = L$, the dispersive flux vanishes:
31//!
32//! $$\left.\frac{\partial u}{\partial x}\right|_{x=L} = 0$$
33//!
34//! Applied as a first-order finite-difference approximation: the last node value
35//! is set equal to its immediate predecessor, enforcing zero gradient.
36//!
37//! ## Souplesse
38//!
39//! The physical parameters $D$ (dispersion) and $v$ (velocity) are struct fields —
40//! they characterise the medium and do not change during a simulation. The feed
41//! concentration is deliberately read from context via a [`ContextVariable`] key,
42//! allowing any dynamics: constant injection, step injection, gradient elution, or
43//! a user-computed profile that itself depends on `Time` or other variables.
44//!
45//! ## Serialisation
46//!
47//! Both types implement `serde::Serialize` / `serde::Deserialize` under the `serde`
48//! feature flag. All fields are either `f64` or [`ContextVariable`] — both
49//! already serialisable.
50//!
51//! [`ComputeContext`]: crate::context::compute::ComputeContext
52//! [`ContextCalculator`]: crate::context::calculator::ContextCalculator
53//! [`ContextVariable`]: crate::context::variable::ContextVariable
54

63// ── DanckwertsInlet ───────────────────────────────────────────────────────────
64
65/// Danckwerts inlet condition — Robin BC at $x = 0$.
66///
67/// Enforces mass-flux continuity at the column inlet:
68///
69/// $$u(0, t) = u_{\text{feed}}(t) + \frac{D}{v}
70///   \left.\frac{\partial u}{\partial x}\right|_{x=0}$$
71///
72/// # Parameters
73///
74/// - `dispersion` — axial dispersion coefficient $D$ $[\text{m}^2 \cdot \text{s}^{-1}]$.
75/// - `velocity` — interstitial velocity $v$ $[\text{m} \cdot \text{s}^{-1}]$.
76/// - `feed_variable` — typed key used to read $u_{\text{feed}}(t)$ from
77///   [`ComputeContext`]. Any `ContextVariable` is accepted — typically an
78///   [`External`](crate::context::variable::ContextVariable::External) variable
79///   produced by a user-supplied calculator (constant, step, gradient elution…).
80///
81/// # Requirements
82///
83/// Declares two required context variables:
84///
85/// - `ContextVariable::SpatialGradient { dimension: 0, component: None }` —
86///   the nodal gradient field $\partial u / \partial x$ computed by a gradient
87///   calculator registered in `SolverConfiguration`.
88/// - `feed_variable` — the feed concentration $u_{\text{feed}}(t)$.
89///
90/// The solver validates that both are covered before the first time step.
91///
92/// # Serialisation
93///
94/// Implements `Serialize` / `Deserialize` under the `serde` feature flag.
95///
96/// # Examples
97///
98/// ```rust
99/// use oxiflow::boundary::danckwerts::DanckwertsInlet;
100/// use oxiflow::context::variable::ContextVariable;
101/// use oxiflow::model::RequiresContext;
102///
103/// let feed_var = ContextVariable::External { name: "feed_concentration".into() };
104/// let inlet = DanckwertsInlet::new(1e-7, 1e-3, feed_var);
105/// assert_eq!(inlet.priority(), 50);
106/// ```
107///
108/// [`ComputeContext`]: crate::context::compute::ComputeContext
109#[non_exhaustive]
110#[derive(Debug, Clone)]
111#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
112pub struct DanckwertsInlet {
113    /// Axial dispersion coefficient $D$.
114    pub dispersion: f64,
115    /// Interstitial velocity $v$.
116    pub velocity: f64,
117    /// Context key used to read the feed concentration $u_{\text{feed}}(t)$.
118    pub feed_variable: ContextVariable,
119}
120
121impl DanckwertsInlet {
122    /// Creates a new `DanckwertsInlet`.
123    ///
124    /// # Parameters
125    ///
126    /// - `dispersion` — axial dispersion coefficient $D$.
127    /// - `velocity`   — interstitial velocity $v$ (must be non-zero at apply time).
128    /// - `feed_variable` — context key for $u_{\text{feed}}(t)$.
129    pub fn new(dispersion: f64, velocity: f64, feed_variable: ContextVariable) -> Self {
130        Self {
131            dispersion,
132            velocity,
133            feed_variable,
134        }
135    }
136}
137
138impl RequiresContext for DanckwertsInlet {
139    fn required_variables(&self) -> Vec<ContextVariable> {
140        vec![
141            ContextVariable::SpatialGradient {
142                dimension: 0,
143                component: None,
144            },
145            self.feed_variable.clone(),
146        ]
147    }
148
149    fn priority(&self) -> u32 {
150        50
151    }
152}
153
154impl BoundaryCondition for DanckwertsInlet {
155    fn boundary_type(&self) -> crate::boundary::BoundaryType {
156        crate::boundary::BoundaryType::Robin
157    }
158
159    fn location(&self) -> Option<crate::boundary::BoundaryLocation> {
160        Some(crate::boundary::BoundaryLocation::Inlet)
161    }
162
163    /// Applies the Danckwerts inlet condition to `state[0]`.
164    ///
165    /// Reads `∂u/∂x|_{x=0}` from the gradient field at node 0 and
166    /// $u_{\text{feed}}$ from the context variable declared at construction.
167    ///
168    /// # Errors
169    ///
170    /// - [`OxiflowError::MissingCalculator`] if the gradient field or the feed
171    ///   variable is absent from the context.
172    /// - [`OxiflowError::TypeMismatch`] if the feed variable is not a `Scalar`.
173    /// - [`OxiflowError::PreconditionFailed`] if the mesh has no nodes or if
174    ///   `velocity` is zero (division by zero in $D/v$).
175    fn apply(
176        &self,
177        state: &mut DVector<f64>,
178        ctx: &ComputeContext,
179        mesh: &dyn Mesh,
180    ) -> Result<(), OxiflowError> {
181        if mesh.n_dof() == 0 {
182            return Err(OxiflowError::PreconditionFailed {
183                context: "DanckwertsInlet",
184                message: "mesh has no degrees of freedom".into(),
185            });
186        }
187        if self.velocity == 0.0 {
188            return Err(OxiflowError::PreconditionFailed {
189                context: "DanckwertsInlet",
190                message: "velocity must be non-zero".into(),
191            });
192        }
193
194        let gradient = ctx.gradient(0)?;
195        let du_dx_0 = gradient[0];
196        let u_feed = ctx.external(self.feed_variable.clone())?.as_scalar()?;
197
198        state[0] = u_feed + (self.dispersion / self.velocity) * du_dx_0;
199        Ok(())
200    }
201}

203// ── DanckwertsOutlet ──────────────────────────────────────────────────────────
204
205/// Danckwerts outlet condition — Neumann BC at $x = L$.
206///
207/// Enforces zero dispersive flux at the column outlet:
208///
209/// $$\left.\frac{\partial u}{\partial x}\right|_{x=L} = 0$$
210///
211/// Applied as a first-order finite-difference approximation:
212///
213/// $$u_{n-1} = u_{n-2}$$
214///
215/// No context variables are required — the condition depends only on the
216/// current field state and the mesh geometry.
217///
218/// # Serialisation
219///
220/// Implements `Serialize` / `Deserialize` under the `serde` feature flag.
221///
222/// # Examples
223///
224/// ```rust
225/// use oxiflow::boundary::danckwerts::DanckwertsOutlet;
226/// use oxiflow::model::RequiresContext;
227///
228/// let outlet = DanckwertsOutlet::new();
229/// assert_eq!(outlet.priority(), 60);
230/// ```
231#[non_exhaustive]
232#[derive(Debug, Clone)]
233#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
234pub struct DanckwertsOutlet;
235
236impl DanckwertsOutlet {
237    /// Creates a new `DanckwertsOutlet`.
238    pub fn new() -> Self {
239        Self
240    }
241}

244    fn default() -> Self {
245        Self::new()
246    }
247}

251        vec![]
252    }

255        60
256    }
257}

259impl BoundaryCondition for DanckwertsOutlet {
260    fn boundary_type(&self) -> crate::boundary::BoundaryType {
261        crate::boundary::BoundaryType::Neumann
262    }
263
264    fn location(&self) -> Option<crate::boundary::BoundaryLocation> {
265        Some(crate::boundary::BoundaryLocation::Outlet)
266    }
267
268    /// Applies the Danckwerts outlet condition.
269    ///
270    /// Sets `state[n-1] = state[n-2]` — first-order approximation of zero
271    /// gradient at the outlet. Requires at least two nodes.
272    ///
273    /// # Errors
274    ///
275    /// - [`OxiflowError::PreconditionFailed`] if the mesh has fewer than two nodes.
276    fn apply(
277        &self,
278        state: &mut DVector<f64>,
279        _ctx: &ComputeContext,
280        mesh: &dyn Mesh,
281    ) -> Result<(), OxiflowError> {
282        let n = mesh.n_dof();
283        if n < 2 {
284            return Err(OxiflowError::PreconditionFailed {
285                context: "DanckwertsOutlet",
286                message: "mesh must have at least 2 nodes".into(),
287            });
288        }
289        state[n - 1] = state[n - 2];
290        Ok(())
291    }
292}

319    fn make_ctx_with_gradient_and_feed(gradient_at_0: f64, u_feed: f64) -> ComputeContext {
320        let mut ctx = ComputeContext::new(0.0, 0.01);
321        let n = 5;
322        let mut grad = DVector::zeros(n);
323        grad[0] = gradient_at_0;
324        ctx.insert(
325            ContextVariable::SpatialGradient {
326                dimension: 0,
327                component: None,
328            },
329            ContextValue::ScalarField(grad),
330        );
331        ctx.insert(feed_var(), ContextValue::Scalar(u_feed));
332        ctx
333    }

423    #[test]
424    fn inlet_apply_formula_is_correct() {
425        // u(0) = u_feed + (D/v) * du_dx_0
426        // With D=1e-4, v=1e-2, du_dx_0=0.5, u_feed=0.8:
427        // u(0) = 0.8 + (1e-4/1e-2)*0.5 = 0.8 + 0.005 = 0.805
428        let inlet = make_inlet(1e-4, 1e-2);
429        let mesh = make_mesh(5);
430        let ctx = make_ctx_with_gradient_and_feed(0.5, 0.8);
431        let mut state = DVector::zeros(5);
432        inlet.apply(&mut state, &ctx, &mesh).unwrap();
433        let expected = 0.8 + (1e-4 / 1e-2) * 0.5;
434        assert!((state[0] - expected).abs() < 1e-12);
435    }
