1//! # Module `solver::methods::step_control`
2//!
3//! Adaptive step-size control — shared machinery for variable-step
4//! integrators (DD-036).
5//!
6//! ## Why this is separate from `dopri45.rs`
7//!
8//! [`StepSizeController`] knows nothing about Runge-Kutta, Butcher
9//! tableaux, or any specific integrator — it takes a single abstract
10//! error norm in and produces an accept/reject decision and a new `dt`
11//! out. [`DoPri45Solver`](super::dopri45::DoPri45Solver) is the first
12//! consumer (the error norm comes from the embedded RK4/5 difference),
13//! but a future adaptive implicit solver (deferred, see DD-033's note on
14//! iterated Newton) would plug in the same controller with a different
15//! error source: the usual truncation-error norm when Newton converges,
16//! or a forced rejection when it doesn't — a guard applied *before*
17//! calling the controller, not a change to the controller itself.
18//!
19//! ## Error norm convention
20//!
21//! Per-component error is scaled against `atol + rtol * |u|` and combined
22//! as an RMS norm — the standard convention (Hairer, Nørsett & Wanner,
23//! *Solving Ordinary Differential Equations I*, §II.4). A norm `<= 1.0`
24//! means the step satisfies the requested tolerance.
25//!
26//! ## Controller
27//!
28//! A PI (proportional-integral) controller (Gustafsson): the new `dt`
29//! depends on both the current error and the previous one, which damps
30//! oscillation in `dt` better than plain proportional control. Falls back
31//! to proportional-only on the very first call (no previous error yet).
32
33use nalgebra::DVector;
34
35/// Adaptive step-size controller — error-source-agnostic (see [module
36/// docs](self)).
37#[derive(Debug, Clone)]
38#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
39pub struct StepSizeController {
40    rtol: f64,
41    atol: f64,
42    dt_min: f64,
43    dt_max: f64,
44    /// Safety factor applied to every suggested `dt` — standard practice
45    /// to bias slightly toward acceptance margin rather than the
46    /// theoretical optimum, which tends to under-predict in practice.
47    safety: f64,
48    /// Order of the embedded error estimator (DoPri45: 4 — the local
49    /// error order of the 4th-order solution used only for error
50    /// estimation; the 5th-order result is what is actually propagated,
51    /// "local extrapolation").
52    order: f64,
53    /// Most recent error norm seen (accepted or rejected attempt) — `None`
54    /// before the first call, in which case `next_dt` falls back to plain
55    /// proportional control.
56    prev_error_norm: Option<f64>,
57}
58
59impl StepSizeController {
60    /// Creates a controller with the standard safety factor (`0.9`).
61    pub fn new(rtol: f64, atol: f64, dt_min: f64, dt_max: f64, order: f64) -> Self {
62        Self {
63            rtol,
64            atol,
65            dt_min,
66            dt_max,
67            safety: 0.9,
68            order,
69            prev_error_norm: None,
70        }
71    }
72
73    /// Combines per-component `error` against `reference` (typically the
74    /// higher-order solution) into a single RMS norm — see [module
75    /// docs](self) for the convention.
76    ///
77    /// `error.len()` and `reference.len()` are assumed equal; mismatched
78    /// lengths are zipped to the shorter one (callers are responsible for
79    /// passing matching vectors — this is an internal helper, not a
80    /// public-API boundary that needs to defend against misuse).
81    pub fn error_norm(&self, error: &DVector<f64>, reference: &DVector<f64>) -> f64 {
82        let n = error.len().max(1);
83        let sum_sq: f64 = error
84            .iter()
85            .zip(reference.iter())
86            .map(|(e, u)| {
87                let scale = self.atol + self.rtol * u.abs();
88                (e / scale).powi(2)
89            })
90            .sum();
91        (sum_sq / n as f64).sqrt()
92    }
93
94    /// `true` if `error_norm` satisfies the tolerance — the step should
95    /// be accepted.
96    pub fn accept(&self, error_norm: f64) -> bool {
97        error_norm <= 1.0
98    }
99
100    /// Computes the next `dt` to try, given the `dt` just attempted and
101    /// the error norm it produced — regardless of whether that attempt is
102    /// accepted or rejected (the caller decides that via
103    /// [`accept`](Self::accept) separately).
104    ///
105    /// Updates the internal PI memory on every call, not only on accepted
106    /// steps: the most recent error measurement is informative even from
107    /// a rejected attempt.
108    ///
109    /// The growth/shrink factor is clamped to `[0.2, 5.0]` (standard
110    /// practice — an unclamped controller can oscillate wildly on rough
111    /// problems), and the resulting `dt` is clamped to `[dt_min, dt_max]`.
112    pub fn next_dt(&mut self, current_dt: f64, error_norm: f64) -> f64 {
113        // Floor to avoid division by zero when the error happens to be
114        // exactly representable as 0.0 (e.g. a perfectly linear problem
115        // hitting machine-exact agreement between the two RK orders).
116        let error_norm = error_norm.max(1e-12);
117
118        let beta1 = 0.7 / self.order;
119        let beta2 = 0.4 / self.order;
120
121        let factor = match self.prev_error_norm {
122            Some(prev) => {
123                let prev = prev.max(1e-12);
124                self.safety * error_norm.powf(-beta1) * prev.powf(beta2)
125            }
126            None => self.safety * error_norm.powf(-1.0 / self.order),
127        };
128
129        let factor = factor.clamp(0.2, 5.0);
130        self.prev_error_norm = Some(error_norm);
131
132        (current_dt * factor).clamp(self.dt_min, self.dt_max)
133    }
134
135    /// The configured minimum step — exposed so callers can detect
136    /// "stuck at dt_min" without re-deriving it.
137    pub fn dt_min(&self) -> f64 {
138        self.dt_min
139    }
140}

144#[cfg(test)]
145mod tests {
146    use super::*;
147
148    fn make_controller() -> StepSizeController {
149        StepSizeController::new(1e-6, 1e-9, 1e-8, 1.0, 4.0)
150    }
151
152    #[test]
153    fn error_norm_zero_when_error_is_zero() {
154        let controller = make_controller();
155        let error = DVector::from_vec(vec![0.0, 0.0, 0.0]);
156        let reference = DVector::from_vec(vec![1.0, 2.0, 3.0]);
157        assert_eq!(controller.error_norm(&error, &reference), 0.0);
158    }
159
160    #[test]
161    fn error_norm_scales_with_tolerance() {
162        let controller = StepSizeController::new(0.0, 1.0, 1e-8, 1.0, 4.0);
163        // rtol=0, atol=1 -> scale is always 1 -> norm = RMS(error) directly.
164        let error = DVector::from_vec(vec![1.0, 1.0]);
165        let reference = DVector::from_vec(vec![100.0, 100.0]); // irrelevant when rtol=0
166        assert!((controller.error_norm(&error, &reference) - 1.0).abs() < 1e-12);
167    }
168
169    #[test]
170    fn accept_at_threshold() {
171        let controller = make_controller();
172        assert!(controller.accept(1.0));
173        assert!(controller.accept(0.5));
174        assert!(!controller.accept(1.0001));
175    }
176
177    #[test]
178    fn next_dt_shrinks_on_large_error() {
179        let mut controller = make_controller();
180        let dt = controller.next_dt(0.1, 100.0); // way over tolerance
181        assert!(dt < 0.1, "expected shrink, got dt={dt}");
182    }
183
184    #[test]
185    fn next_dt_grows_on_small_error() {
186        let mut controller = make_controller();
187        let dt = controller.next_dt(0.1, 0.01); // well within tolerance
188        assert!(dt > 0.1, "expected growth, got dt={dt}");
189    }
190
191    #[test]
192    fn next_dt_respects_dt_max_clamp() {
193        let mut controller = StepSizeController::new(1e-6, 1e-9, 1e-8, 0.2, 4.0);
194        let dt = controller.next_dt(0.1, 1e-9); // tiny error -> wants to grow a lot
195        assert!(dt <= 0.2, "expected clamp to dt_max=0.2, got dt={dt}");
196    }
197
198    #[test]
199    fn next_dt_respects_dt_min_clamp() {
200        let mut controller = StepSizeController::new(1e-6, 1e-9, 0.05, 1.0, 4.0);
201        let dt = controller.next_dt(0.1, 1e6); // huge error -> wants to shrink a lot
202        assert!(dt >= 0.05, "expected clamp to dt_min=0.05, got dt={dt}");
203    }
204
205    #[test]
206    fn next_dt_growth_factor_is_bounded() {
207        let mut controller = make_controller();
208        // Even with a vanishingly small error, growth in one step is
209        // capped at 5x -- the clamp documented above.
210        let dt = controller.next_dt(0.1, 1e-12);
211        assert!(
212            dt <= 0.1 * 5.0 + 1e-12,
213            "expected growth capped at 5x, got dt={dt}"
214        );
215    }
216
217    #[test]
218    fn dt_min_accessor_matches_constructor() {
219        let controller = StepSizeController::new(1e-6, 1e-9, 1e-7, 1.0, 4.0);
220        assert_eq!(controller.dt_min(), 1e-7);
221    }
222
223    #[test]
224    fn pi_memory_affects_second_call() {
225        // Same error norm twice in a row -- the second call's factor
226        // differs from the first because the PI term now has a previous
227        // error to compare against (None -> Some transition).
228        let mut controller = make_controller();
229        let dt1 = controller.next_dt(0.1, 0.5);
230        let dt2 = controller.next_dt(dt1, 0.5);
231        // Not asserting a specific direction -- just that the PI memory
232        // actually changes the computation path (first call uses the
233        // P-only fallback, second uses the full PI formula).
234        assert!(controller.prev_error_norm.is_some());
235        let _ = dt2; // exercised for the side effect on prev_error_norm
236    }
237
238    // ── Serde round-trip (#70) ──────────────────────────────────────────────
239
240    #[cfg(feature = "serde")]
241    #[test]
242    fn serde_roundtrip_preserves_parameters() {
243        let mut controller = make_controller();
244        // Exercise prev_error_norm: Some(_), not just the just-constructed
245        // None state -- the round-trip must preserve it either way.
246        let _ = controller.next_dt(0.1, 0.5);
247
248        let json = serde_json::to_string(&controller).unwrap();
249        let restored: StepSizeController = serde_json::from_str(&json).unwrap();
250
251        assert_eq!(restored.dt_min(), controller.dt_min());
252        assert_eq!(restored.prev_error_norm, controller.prev_error_norm);
253    }
