1//! # Module `solver::methods`
2//!
3//! Temporal integration methods.
4//!
5//! ## Active at J4a (v0.4.0)
6//!
7//! | Method | Type | Issue |
8//! |---|---|---|
9//! | [`euler::ForwardEulerSolver`] | Explicit, 1st order | #33, #41 |
10//! | [`rk4::RK4Solver`] | Explicit, 4th order | #41 |
11//! | [`backward_euler::BackwardEulerSolver`] | Implicit, 1st order | #43, DD-013, DD-033 |
12//! | [`crank_nicolson::CrankNicolsonSolver`] | Semi-implicit, 2nd order | #43, DD-013, DD-033 |
13//! | [`bdf2::BDF2Solver`] | Implicit multi-step, 2nd order | #44, DD-034 |
14//! | [`dopri45::DoPri45Solver`] | Adaptive explicit, order 5 | #42, DD-036 |
15//! | [`imex::OperatorSplittingSolver`] | Operator splitting (Strang), n ≥ 2 | #45, DD-037 |
16//!
17//! All integrators are decoupled from the spatial scheme via
18//! `DiscreteOperator<M: Mesh>` (INV-2, J4b) — no FD/FV/FEM
19//! method is called directly inside an integrator.
20//!
21//! ## Shared evaluation contract
22//!
23//! Every integrator — single-stage (Euler) or multi-stage (RK4 and beyond) —
24//! follows the same per-evaluation order documented in
25//! [`crate::solver`]: calculators, then boundary conditions, then
26//! `compute_physics`. [`evaluate_derivative`] implements this contract once
27//! so each solver doesn't reimplement it per stage.
28//!
29//! Implicit methods ([`backward_euler`], [`crank_nicolson`], [`bdf2`]) build
30//! on the same [`evaluate_derivative`] for their explicit-derivative
31//! evaluation, plus the shared machinery in [`implicit`] (frozen-Jacobian
32//! Newton correction, DD-033) for the implicit part.
33//!
34//! ## Per-step primitive, with history (DD-031, DD-034)
35//!
36//! [`SteppableSolver`] exposes a single-step primitive on top of `Solver`'s
37//! full-time-range `solve()`. [`crate::solver::orchestrator::MultiDomainOrchestrator`]
38//! calls it once per domain per synchronised step, so each domain in a
39//! coupled scenario can use a different integrator. Multi-step methods
40//! ([`bdf2::BDF2Solver`], needing `u^{n-1}`) declare
41//! [`SteppableSolver::history_depth`] — callers maintain a buffer of that
42//! many past states and pass it to [`SteppableSolver::step`]. One-step
43//! methods use the default (`0`) and ignore the parameter (DD-034).
44//!
45//! [`SteppableSolver::solve_fixed_step`] (DD-035) factors out the common
46//! `solve()` body every fixed-step solver above used to duplicate
47//! near-verbatim — each one's `Solver::solve()` is now a single call to
48//! it. [`dopri45::DoPri45Solver`] does **not** use this: variable-step
49//! methods need their own loop (`dt` changes every iteration, no fixed
50//! total step count) and implement `Solver` directly instead — see
51//! [`step_control`] (DD-036) for the shared piece adaptive methods *do*
52//! get: the step-size controller itself, decoupled from any specific
53//! integrator's error source.
54//!
55//! [`imex::OperatorSplittingSolver`] (DD-037, #45) also implements `Solver`
56//! directly rather than `SteppableSolver` — same posture as DoPri45: each
57//! of its sub-operators already carries its own `Domain`, so fitting it
58//! into the single-`Domain`-per-step shape of `SteppableSolver::step`
59//! would reopen the multi-domain orchestration question for a need #45
60//! never asked for.

92/// Applies every boundary condition attached to `domain` to `state`, in
93/// declaration order.
94///
95/// Called once per derivative evaluation, after context calculators and
96/// before `compute_physics` — see [`crate::boundary::BoundaryCondition::apply`].
97/// For multi-stage integrators this means once *per stage*, on that stage's
98/// intermediate state: a Dirichlet value must be re-enforced at every
99/// evaluation point, not just once per outer time step, or it would drift
100/// across stages.
101///
102/// No-op if `domain` has no boundary conditions attached (J1-style scenarios).
103pub(crate) fn apply_boundary_conditions(
104    domain: &Domain,
105    state: &mut ContextValue,
106    ctx: &ComputeContext,
107) -> Result<(), OxiflowError> {
108    if domain.boundary_conditions.is_empty() {
109        return Ok(());
110    }
111    let field = state.as_scalar_field_mut()?;
112    for bc in &domain.boundary_conditions {
113        bc.apply(field, ctx, domain.mesh.as_ref())?;
114    }
115    Ok(())
116}

118/// Evaluates `du/dt` for `domain` at time `t`, following the contractual
119/// order: build `ComputeContext`, run the calculator chain, apply boundary
120/// conditions, then call `compute_physics`.
121///
122/// `state` is mutated in-place by boundary condition application — this is
123/// intentional (see [`apply_boundary_conditions`]) and mirrors the contract
124/// documented on [`crate::boundary::BoundaryCondition::apply`]. Callers pass
125/// either the persisted solution state (single-stage methods) or a
126/// transient intermediate stage state (multi-stage methods); in both cases
127/// the corrected state is also what `compute_physics` consumes.
128pub(crate) fn evaluate_derivative(
129    domain: &Domain,
130    chain: &[&dyn ContextCalculator],
131    state: &mut ContextValue,
132    t: f64,
133    dt: f64,
134) -> Result<ContextValue, OxiflowError> {
135    let mut ctx = ComputeContext::new(t, dt);
136
137    for calc in chain {
138        let value = calc
139            .compute(state, &ctx)
140            .map_err(|e| OxiflowError::ComputationFailed {
141                variable: calc.provides(),
142                source: Box::new(e),
143            })?;
144        ctx.insert(calc.provides(), value);
145    }
146
147    apply_boundary_conditions(domain, state, &ctx)?;
148
149    domain.model.compute_physics(state, &ctx)
150}

152/// Checks that all nodal values in `state` are finite.
153///
154/// Shared between integrators — returns `OxiflowError::SolverDivergence` on
155/// the first non-finite value found.
156pub(crate) fn check_finite(state: &ContextValue, t: f64) -> Result<(), OxiflowError> {
157    if let Ok(field) = state.as_scalar_field() {
158        if field.iter().any(|v| !v.is_finite()) {
159            return Err(OxiflowError::SolverDivergence {
160                time: t,
161                reason: "non-finite value detected in state vector".into(),
162            });
163        }
164    }
165    Ok(())
166}

170/// Single-domain solvers that expose a per-step primitive (DD-031, DD-034).
171///
172/// `Solver::solve()` drives a full time range for one domain; `step()`
173/// advances a single domain by exactly one `dt`, given its current state
174/// and (for multi-step methods) a bounded history of past states. This is
175/// what [`crate::solver::orchestrator::MultiDomainOrchestrator`] calls
176/// once per domain per synchronised step, allowing each domain in a
177/// coupled scenario to use a different integrator.
178///
179/// Implementations should extract their existing single-step logic from
180/// `solve()` rather than duplicate it — `solve()` is expected to call
181/// `step()` internally.
182pub trait SteppableSolver: Solver {
183    /// Number of past states (beyond the current one) this solver needs.
184    ///
185    /// `0` for one-step methods (Euler, RK4, Backward Euler,
186    /// Crank-Nicolson) — the default. `1` for BDF2 (needs `u^{n-1}`).
187    /// Callers (the per-solver `solve()` loop, or
188    /// [`crate::solver::orchestrator::MultiDomainOrchestrator`]) are
189    /// responsible for maintaining a buffer of exactly this many past
190    /// states and passing it to [`step`](Self::step).
191    fn history_depth(&self) -> usize {
192        0
193    }
194
195    /// Advances `state` by one step of size `dt`, at time `t`, for `domain`.
196    ///
197    /// `chain` is the calculator chain already built for this run (built
198    /// once by the caller, reused across steps — see
199    /// [`crate::solver::chain::build_calculator_chain`]).
200    ///
201    /// `history` holds up to [`history_depth`](Self::history_depth) past
202    /// states, most recent first (`history[0]` is `u^{n-1}`) — shorter
203    /// than that during startup (empty on the very first step). Solvers
204    /// with `history_depth() == 0` ignore this parameter entirely.
205    ///
206    /// `state` may be mutated in-place by boundary condition application
207    /// (see [`evaluate_derivative`]); the returned value is the state at
208    /// `t + dt`.
209    fn step(
210        &self,
211        domain: &Domain,
212        chain: &[&dyn ContextCalculator],
213        state: &mut ContextValue,
214        history: &[ContextValue],
215        t: f64,
216        dt: f64,
217    ) -> Result<ContextValue, OxiflowError>;
218
219    /// Drives a full fixed-step time range, calling [`step`](Self::step)
220    /// in a loop and managing the history buffer generically via
221    /// [`history_depth`](Self::history_depth) — the common body every
222    /// fixed-step solver (Euler, RK4, Backward Euler, Crank-Nicolson,
223    /// BDF2) used to duplicate near-verbatim.
224    ///
225    /// `Solver::solve()` implementations for fixed-step methods are
226    /// expected to be a single call to this: `self.solve_fixed_step(scenario, config)`.
227    ///
228    /// Variable-step methods (DoPri45) do **not** use this — they need
229    /// their own loop (`dt` changes every iteration, total step count
230    /// isn't known upfront) and implement `Solver::solve()` directly
231    /// instead.
232    ///
233    /// # Errors
234    ///
235    /// `OxiflowError::InvalidDomain` if `config.time.step_control` is not
236    /// `StepControl::Fixed`, if `dt <= 0.0`, or if `t_end <= t_start`.
237    fn solve_fixed_step(
238        &self,
239        scenario: &Scenario,
240        config: &SolverConfiguration,
241    ) -> Result<SimulationResult, OxiflowError> {
242        scenario.validate()?;
243        let domain = scenario.single_domain()?;
244
245        let dt = match &config.time.step_control {
246            StepControl::Fixed { dt } => *dt,
247            _ => {
248                return Err(OxiflowError::InvalidDomain(
249                    "this solver only supports StepControl::Fixed (adaptive step control \
250                     not supported by this method)"
251                        .into(),
252                ))
253            }
254        };
255
256        let t_end = config.time.t_end;
257        let t_start = scenario.t_start;
258
259        if dt <= 0.0 {
260            return Err(OxiflowError::InvalidDomain(
261                "dt must be strictly positive".into(),
262            ));
263        }
264        if t_end <= t_start {
265            return Err(OxiflowError::InvalidDomain(
266                "t_end must be greater than t_start".into(),
267            ));
268        }
269
270        let requirements = scenario.context_requirements();
271        let chain = build_calculator_chain(&requirements, &config.calculators)?;
272
273        let mut u = domain.model.initial_state(domain.mesh.as_ref());
274        let mut history: Vec<ContextValue> = Vec::new();
275
276        let n_steps = ((t_end - t_start) / dt).round() as usize;
277        let save_every = config.time.save_every.unwrap_or(1);
278        let capacity = n_steps / save_every + 1;
279        let mut states: Vec<ContextValue> = Vec::with_capacity(capacity);
280        let mut times: Vec<f64> = Vec::with_capacity(capacity);
281
282        states.push(u.clone());
283        times.push(t_start);
284
285        let depth = self.history_depth();
286
287        for step in 0..n_steps {
288            let t = t_start + (step as f64) * dt;
289            let t_next = t_start + ((step + 1) as f64) * dt;
290
291            let next = self.step(domain, &chain, &mut u, &history, t, dt)?;
292
293            if depth > 0 {
294                // `u` was mutated in-place by BC application inside
295                // `step()` above -- push *that* corrected value, not a
296                // pre-correction one. `mem::replace` avoids a clone: `u`
297                // is moved into history, `next` takes its place.
298                let prev = std::mem::replace(&mut u, next);
299                history.insert(0, prev);
300                history.truncate(depth);
301            } else {
302                u = next;
303            }
304
305            check_finite(&u, t_next)?;
306
307            if (step + 1) % save_every == 0 {
308                states.push(u.clone());
309                times.push(t_next);
310            }
311        }
312
313        Ok(SimulationResult {
314            states,
315            times,
316            n_steps,
317            metadata: HashMap::new(),
318        })
319    }
