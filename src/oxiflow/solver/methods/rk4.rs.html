1//! # Module `solver::methods::rk4`
2//!
3//! Classical Runge-Kutta 4 integrator — explicit, 4th order (issue #41).
4//!
5//! ## Algorithm
6//!
7//! At each time step, four stage derivatives are evaluated:
8//!
9//! $$
10//! \begin{aligned}
11//! k_1 &= f(t,\ u) \\
12//! k_2 &= f(t + \tfrac{\Delta t}{2},\ u + \tfrac{\Delta t}{2} k_1) \\
13//! k_3 &= f(t + \tfrac{\Delta t}{2},\ u + \tfrac{\Delta t}{2} k_2) \\
14//! k_4 &= f(t + \Delta t,\ u + \Delta t\, k_3)
15//! \end{aligned}
16//! $$
17//!
18//! combined into the step:
19//!
20//! $$u^{n+1} = u^n + \frac{\Delta t}{6}\left(k_1 + 2k_2 + 2k_3 + k_4\right)$$
21//!
22//! where $f = \text{compute\_physics}(u, \text{ctx})$, evaluated on a state
23//! that has had boundary conditions enforced for *that* stage — see
24//! [`super::evaluate_derivative`].
25//!
26//! ## Boundary conditions across stages
27//!
28//! Each of the four stages is a separate derivative evaluation, so each one
29//! gets its own boundary-condition application via
30//! [`super::apply_boundary_conditions`] — on `u` itself for stage 1, and on
31//! the transient intermediate states (`u + scale * k_i`) for stages 2-4. A
32//! Dirichlet value enforced only once per outer step would otherwise drift
33//! across the intermediate evaluations.
34//!
35//! `t` passed to [`crate::context::compute::ComputeContext`] varies per
36//! stage (`t`, `t + dt/2`, `t + dt/2`, `t + dt`); `dt` itself is the
37//! *outer* step size for all four stages — it identifies the configured
38//! step, not a per-stage time delta.
39//!
40//! ## Scope at J4a
41//!
42//! - Single-domain scenarios only (`n_domains() == 1`) — same restriction
43//!   as `ForwardEulerSolver`. Multi-domain `CouplingOperator` scenarios are
44//!   out of scope; see #40 for the dedicated multi-domain proto.
45//! - No `DiscreteOperator` (INV-2) — spatial schemes arrive at J4b.
46//! - `StepControl::Fixed { dt }` only — adaptive step control is DoPri45's
47//!   job (reserved, J4).
48//!
49//! ## Stability
50//!
51//! Classical RK4 has a larger stability region than forward Euler along the
52//! imaginary axis, but is still explicit: the CFL condition
53//! $\text{CFL} = v\,\Delta t / \Delta x \leq 1$ still bounds the usable
54//! step size. The solver does not enforce this automatically.

65/// Classical Runge-Kutta 4 solver — explicit, 4th order.
66///
67/// Implements the `Solver` trait for single-domain problems with fixed step
68/// control. See [module documentation](self) for algorithm details.
69///
70/// # Examples
71///
72/// ```rust,ignore
73/// use oxiflow::solver::methods::rk4::RK4Solver;
74/// use oxiflow::solver::{Scenario, SolverConfiguration, TimeConfiguration, StepControl, IntegratorKind};
75///
76/// let scenario = Scenario::single(Box::new(my_model), Box::new(mesh));
77/// let config = SolverConfiguration::new(
78///     TimeConfiguration::new(100.0, StepControl::Fixed { dt: 0.1 }),
79///     IntegratorKind::RK4,
80/// );
81/// let result = RK4Solver.solve(&scenario, &config).unwrap();
82/// ```
83pub struct RK4Solver;
84
85impl Solver for RK4Solver {
86    fn solve(
87        &self,
88        scenario: &Scenario,
89        config: &SolverConfiguration,
90    ) -> Result<SimulationResult, OxiflowError> {
91        self.solve_fixed_step(scenario, config)
92    }
93}

95impl SteppableSolver for RK4Solver {
96    fn step(
97        &self,
98        domain: &Domain,
99        chain: &[&dyn ContextCalculator],
100        state: &mut ContextValue,
101        _history: &[ContextValue],
102        t: f64,
103        dt: f64,
104    ) -> Result<ContextValue, OxiflowError> {
105        // history_depth() defaults to 0 -- RK4 is a one-step method,
106        // `_history` is always empty here and intentionally unused.
107        let half_dt = dt / 2.0;
108
109        // Stage 1: k1 = f(t, u). BCs are applied to `state` itself here —
110        // it is the persisted solution state, same contract as Euler.
111        let k1 = evaluate_derivative(domain, chain, state, t, dt)?;
112
113        // Stage 2: k2 = f(t + dt/2, u + dt/2 * k1)
114        let mut u2 = combine(state, &k1, half_dt)?;
115        let k2 = evaluate_derivative(domain, chain, &mut u2, t + half_dt, dt)?;
116
117        // Stage 3: k3 = f(t + dt/2, u + dt/2 * k2)
118        let mut u3 = combine(state, &k2, half_dt)?;
119        let k3 = evaluate_derivative(domain, chain, &mut u3, t + half_dt, dt)?;
120
121        // Stage 4: k4 = f(t + dt, u + dt * k3)
122        let mut u4 = combine(state, &k3, dt)?;
123        let k4 = evaluate_derivative(domain, chain, &mut u4, t + dt, dt)?;
124
125        // Weighted combination: u_next = u + dt/6 * (k1 + 2k2 + 2k3 + k4)
126        rk4_combine(state, &k1, &k2, &k3, &k4, dt)
127    }
128}

130/// Computes `u + scale * k` for `ScalarField` states.
131///
132/// Returns `OxiflowError::TypeMismatch` if either operand is not
133/// `ScalarField`, or `InvalidDomain` if their lengths differ.
134fn combine(u: &ContextValue, k: &ContextValue, scale: f64) -> Result<ContextValue, OxiflowError> {
135    let u_field = u.as_scalar_field()?;
136    let k_field = k.as_scalar_field()?;
137
138    if u_field.len() != k_field.len() {
139        return Err(OxiflowError::InvalidDomain(format!(
140            "state length {} != stage derivative length {}",
141            u_field.len(),
142            k_field.len()
143        )));
144    }
145
146    Ok(ContextValue::ScalarField(u_field + k_field * scale))
147}
148
149/// Computes the RK4 weighted combination `u + dt/6 * (k1 + 2k2 + 2k3 + k4)`.
150///
151/// Returns `OxiflowError::InvalidDomain` if any stage derivative's length
152/// differs from `u`'s.
153fn rk4_combine(
154    u: &ContextValue,
155    k1: &ContextValue,
156    k2: &ContextValue,
157    k3: &ContextValue,
158    k4: &ContextValue,
159    dt: f64,
160) -> Result<ContextValue, OxiflowError> {
161    let u_field = u.as_scalar_field()?;
162    let k1_field = k1.as_scalar_field()?;
163    let k2_field = k2.as_scalar_field()?;
164    let k3_field = k3.as_scalar_field()?;
165    let k4_field = k4.as_scalar_field()?;
166
167    let n = u_field.len();
168    if k1_field.len() != n || k2_field.len() != n || k3_field.len() != n || k4_field.len() != n {
169        return Err(OxiflowError::InvalidDomain(
170            "RK4 stage derivative length mismatch with state".into(),
171        ));
172    }
173
174    // Use fully-owned DVector operands throughout: avoids relying on
175    // nalgebra's reference/owned operator-overload resolution across a long
176    // expression chain — every `+`/`*` here is owned-op-owned, the most
177    // basic and unambiguous combination.
178    let k1_owned = k1_field.clone();
179    let k2_owned = k2_field.clone();
180    let k3_owned = k3_field.clone();
181    let k4_owned = k4_field.clone();
182
183    let weighted: DVector<f64> = k1_owned + k2_owned * 2.0 + k3_owned * 2.0 + k4_owned;
184    let result: DVector<f64> = u_field.clone() + weighted * (dt / 6.0);
185
186    Ok(ContextValue::ScalarField(result))
187}

396    #[test]
397    fn rk4_error_reduces_by_16x_with_step_halving() {
398        // 4th-order method: halving dt should reduce the error by ~2^4 = 16x.
399        let lambda: f64 = 1.0;
400        let t_end: f64 = 1.0;
401        let analytical = (-(lambda * t_end)).exp();
402
403        let error_at = |dt: f64| -> f64 {
404            let scenario = Scenario::single(Box::new(ExponentialDecay { lambda }), make_mesh(2));
405            let config = make_config(t_end, dt);
406            let result = RK4Solver.solve(&scenario, &config).unwrap();
407            let val = result.states.last().unwrap().as_scalar_field().unwrap()[0];
408            (val - analytical).abs()
409        };
410
411        // Coarser steps than Euler's order test: RK4 error at very small dt
412        // approaches f64 round-off, which would mask the 4th-order rate.
413        let error_coarse = error_at(0.1);
414        let error_fine = error_at(0.05);
415        let ratio = error_coarse / error_fine;
416
417        assert!(
418            (13.0..19.0).contains(&ratio),
419            "expected ~16x error reduction on dt halving, got {:.3}x (coarse={:.2e}, fine={:.2e})",
420            ratio,
421            error_coarse,
422            error_fine
423        );
424    }

460    #[test]
461    fn boundary_condition_is_applied_at_every_stage() {
462        use crate::boundary::{BoundaryCondition, BoundaryType};
463        use crate::mesh::Mesh as MeshTrait;
464        use crate::solver::scenario::Domain;
465        use std::sync::atomic::{AtomicUsize, Ordering};
466        use std::sync::Arc;
467
468        /// Pins node 0 and counts how many times it was applied — used to
469        /// confirm RK4 applies BCs once per stage (4 per step), not once
470        /// per outer step.
471        ///
472        /// `Arc<AtomicUsize>`, not `Rc<Cell<usize>>` — `BoundaryCondition`
473        /// requires `Send + Sync` since DD-037 (#45): a `Domain` can now be
474        /// *owned* by `OperatorSplittingSolver`'s `SplitOperator`, which
475        /// must itself be `Send + Sync` (`Solver: Send + Sync`).
476        #[derive(Debug)]
477        struct PinFirstNode {
478            value: f64,
479            calls: Arc<AtomicUsize>,
480        }
481
482        impl RequiresContext for PinFirstNode {
483            fn required_variables(&self) -> Vec<ContextVariable> {
484                vec![]
485            }
486        }
487
488        impl BoundaryCondition for PinFirstNode {
489            fn boundary_type(&self) -> BoundaryType {
490                BoundaryType::Dirichlet
491            }
492
493            fn apply(
494                &self,
495                state: &mut DVector<f64>,
496                _ctx: &ComputeContext,
497                _mesh: &dyn MeshTrait,
498            ) -> Result<(), OxiflowError> {
499                state[0] = self.value;
500                self.calls.fetch_add(1, Ordering::SeqCst);
501                Ok(())
502            }
503        }
504
505        let calls = Arc::new(AtomicUsize::new(0));
506        let domain = Domain::new("pinned", Box::new(ZeroDerivative), make_mesh(3))
507            .with_boundary_conditions(vec![Box::new(PinFirstNode {
508                value: -7.0,
509                calls: calls.clone(),
510            })]);
511        let scenario = Scenario::multi(vec![domain]).unwrap();
512        let config = make_config(0.2, 0.1); // 2 steps
513
514        let result = RK4Solver.solve(&scenario, &config).unwrap();
515
516        let final_state = result.states.last().unwrap().as_scalar_field().unwrap();
517        assert!((final_state[0] - (-7.0)).abs() < 1e-12);
518        assert!((final_state[1] - 2.5).abs() < 1e-12);
519
520        // 2 steps * 4 stages = 8 applications.
521        assert_eq!(calls.load(Ordering::SeqCst), 8);
522    }

594    fn rk4_combine_reduces_to_euler_when_stages_are_equal() {
595        // If k1 == k2 == k3 == k4 == k, the weighted sum (k+2k+2k+k)/6 = k,
596        // so RK4's update degenerates to the Euler update u + dt*k.
597        let u = ContextValue::ScalarField(DVector::from_vec(vec![1.0, 2.0]));
598        let k = ContextValue::ScalarField(DVector::from_vec(vec![0.5, -0.5]));
599        let dt = 0.2;
600
601        let rk4_result = rk4_combine(&u, &k, &k, &k, &k, dt).unwrap();
602        let euler_result = combine(&u, &k, dt).unwrap();
603
604        let rk4_field = rk4_result.as_scalar_field().unwrap();
605        let euler_field = euler_result.as_scalar_field().unwrap();
606        for i in 0..2 {
607            assert!((rk4_field[i] - euler_field[i]).abs() < 1e-12);
608        }
609    }
