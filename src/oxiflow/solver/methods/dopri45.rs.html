1//! # Module `solver::methods::dopri45`
2//!
3//! Dormand-Prince DoPri45 — explicit, adaptive step, order 5 (local
4//! extrapolation) with embedded order-4 error estimate (issue #42).
5//!
6//! ## Algorithm
7//!
8//! 7-stage embedded Runge-Kutta pair. The FSAL (First Same As Last)
9//! property of the Dormand-Prince tableau means stage 7's evaluation
10//! point **is** the 5th-order solution — no separate weighted sum needed
11//! to obtain it (see `dopri45_stages` below). The 4th-order solution is
12//! never computed explicitly either: only the *difference*
13//! `b_i - b*_i` is needed for the error estimate.
14//!
15//! Butcher tableau coefficients verified by hand before writing this file
16//! (no compiler available in the authoring environment): row sums
17//! `sum_j a_ij = c_i` for every stage, `sum(b) = 1`, `sum(b*) = 1`. These
18//! are necessary consistency conditions for any valid RK tableau — they
19//! don't guarantee every individual coefficient is correct, but they
20//! would catch most transcription errors. Cross-check against a
21//! published source (Hairer/Nørsett/Wanner, or Dormand & Prince 1980)
22//! before trusting this in production.
23//!
24//! ## Step-size control (DD-036)
25//!
26//! Delegates entirely to [`StepSizeController`](super::step_control::StepSizeController)
27//! for the accept/reject decision and the next `dt` — this file only
28//! supplies the error norm input (the RK4/5 difference) and the rejection
29//! retry loop. See [`super::step_control`] for the controller itself.
30//!
31//! ## Scope — `Solver` only, not `SteppableSolver` (see discussion)
32//!
33//! Unlike BDF2 (DD-034), this solver does **not** implement
34//! `SteppableSolver`. BDF2's gap (needing `u^{n-1}`) was orthogonal to
35//! `dt` itself; this solver's defining feature is choosing its *own*
36//! `dt` across calls, which is in direct tension with
37//! `MultiDomainOrchestrator`'s v1 scope (DD-031: `dt` synchronised across
38//! domains). Making this orchestrator-compatible would mean re-opening
39//! the multirate question DD-031 already deferred — not a gap orthogonal
40//! to an existing limitation, but the same one. Revisit together if
41//! multirate coupling is ever tackled.
42//!
43//! ## Acceptance criteria mapping (#42)
44//!
45//! - Step-size control within `rtol`/`atol`: enforced by
46//!   `StepSizeController::accept`.
47//! - `dt_min` guard -> `OxiflowError::SolverDivergence`: the rejection
48//!   retry loop below bails out once the controller can no longer
49//!   suggest a smaller `dt`.
50//! - Accepted/rejected counts in `SimulationResult::metadata`: keys
51//!   `"solver.accepted_steps"` / `"solver.rejected_steps"`, following the
52//!   convention already documented on `SimulationResult`.
53//! - Performance comparable to fixed-step RK4 on non-stiff problems: not
54//!   benchmarked here (no compiler in this environment) — worth checking
55//!   once this runs for real.

98const A65: f64 = -5103.0 / 18656.0;

100const A71: f64 = 35.0 / 384.0;
101// A72 = 0.0 -- the k2 term drops out of the stage-7 (and 5th-order
102// solution) combination entirely.
103const A73: f64 = 500.0 / 1113.0;
104const A74: f64 = 125.0 / 192.0;
105const A75: f64 = -2187.0 / 6784.0;
106const A76: f64 = 11.0 / 84.0;

108/// Error weights `b_i - b*_i` (5th-order minus 4th-order), precomputed by
109/// hand (see module docs for the derivation and consistency check).
110const E1: f64 = 71.0 / 57600.0;
111const E2: f64 = 0.0;
112const E3: f64 = -71.0 / 16695.0;
113const E4: f64 = 71.0 / 1920.0;
114const E5: f64 = -17253.0 / 339200.0;
115const E6: f64 = 22.0 / 525.0;
116const E7: f64 = -1.0 / 40.0;

118/// Order of the embedded (lower-order) error estimator -- used by the
119/// step-size controller's exponents (DD-036).
120const ERROR_ESTIMATOR_ORDER: f64 = 4.0;

122/// Safety guard against an infinite rejection loop on a single step --
123/// the controller-driven `dt`-shrink should converge well before this in
124/// any well-posed problem; hitting it means tolerance genuinely cannot be
125/// satisfied (acceptance criterion, #42).
126const MAX_REJECTIONS_PER_STEP: usize = 50;

128/// Dormand-Prince DoPri45 solver — explicit, adaptive step.
129///
130/// See [module docs](self) for the algorithm, the step-size controller
131/// delegation, and why this implements `Solver` only (not
132/// `SteppableSolver`).
133pub struct DoPri45Solver;

135impl Solver for DoPri45Solver {
136    fn solve(
137        &self,
138        scenario: &Scenario,
139        config: &SolverConfiguration,
140    ) -> Result<SimulationResult, OxiflowError> {
141        scenario.validate()?;
142        let domain = scenario.single_domain()?;
143
144        let (dt_init, dt_min, dt_max, rtol, atol) = match &config.time.step_control {
145            StepControl::Adaptive {
146                dt_init,
147                dt_min,
148                dt_max,
149                rtol,
150                atol,
151            } => (*dt_init, *dt_min, *dt_max, *rtol, *atol),
152            _ => {
153                return Err(OxiflowError::InvalidDomain(
154                    "DoPri45Solver only supports StepControl::Adaptive".into(),
155                ))
156            }
157        };
158
159        let t_end = config.time.t_end;
160        let t_start = scenario.t_start;
161
162        if dt_init <= 0.0 || dt_min <= 0.0 || dt_max < dt_min {
163            return Err(OxiflowError::InvalidDomain(
164                "dt_init and dt_min must be strictly positive, and dt_max >= dt_min".into(),
165            ));
166        }
167        if t_end <= t_start {
168            return Err(OxiflowError::InvalidDomain(
169                "t_end must be greater than t_start".into(),
170            ));
171        }
172
173        let requirements = scenario.context_requirements();
174        let chain = build_calculator_chain(&requirements, &config.calculators)?;
175
176        let mut u = domain.model.initial_state(domain.mesh.as_ref());
177        let mut t = t_start;
178        let mut dt = dt_init;
179
180        let mut controller =
181            StepSizeController::new(rtol, atol, dt_min, dt_max, ERROR_ESTIMATOR_ORDER);
182
183        // `save_every` counts *accepted* steps -- there's no fixed total
184        // step count to divide by here, unlike `solve_fixed_step`.
185        let save_every = config.time.save_every.unwrap_or(1);
186        let mut accepted_since_save = 0usize;
187
188        let mut states: Vec<ContextValue> = vec![u.clone()];
189        let mut times: Vec<f64> = vec![t_start];
190
191        let mut accepted_steps = 0usize;
192        let mut rejected_steps = 0usize;
193
194        // Small tolerance against floating-point accumulation in `t`
195        // (unavoidable here, unlike the fixed-step solvers: `dt` varies,
196        // so there's no `t_start + step * dt` closed form to fall back
197        // on -- see euler.rs's module docs for that fix, which doesn't
198        // apply to variable-step methods).
199        while t < t_end - 1e-12 {
200            let mut local_dt = dt.min(t_end - t);
201            let mut rejections_this_step = 0usize;
202
203            loop {
204                let mut attempt_state = u.clone();
205                let (y5_field, error_field) =
206                    dopri45_stages(domain, &chain, &mut attempt_state, t, local_dt)?;
207
208                let error_norm = controller.error_norm(&error_field, &y5_field);
209
210                if controller.accept(error_norm) {
211                    u = ContextValue::ScalarField(y5_field);
212                    t += local_dt;
213                    accepted_steps += 1;
214                    accepted_since_save += 1;
215
216                    check_finite(&u, t)?;
217
218                    dt = controller.next_dt(local_dt, error_norm);
219
220                    if accepted_since_save >= save_every {
221                        states.push(u.clone());
222                        times.push(t);
223                        accepted_since_save = 0;
224                    }
225                    break;
226                }
227
228                rejected_steps += 1;
229                rejections_this_step += 1;
230                let suggested = controller.next_dt(local_dt, error_norm);
231
232                if rejections_this_step >= MAX_REJECTIONS_PER_STEP
233                    || suggested <= controller.dt_min()
234                {
235                    return Err(OxiflowError::SolverDivergence {
236                        time: t,
237                        reason: format!(
238                            "step rejected {rejections_this_step} times in a row; cannot \
239                             satisfy rtol/atol even at dt_min={}",
240                            controller.dt_min()
241                        ),
242                    });
243                }
244
245                local_dt = suggested;
246            }
247        }
248
249        let mut metadata = HashMap::new();
250        metadata.insert("solver.accepted_steps".to_string(), accepted_steps as f64);
251        metadata.insert("solver.rejected_steps".to_string(), rejected_steps as f64);
252
253        Ok(SimulationResult {
254            states,
255            times,
256            n_steps: accepted_steps,
257            metadata,
258        })
259    }
260}

262/// Computes the 7 Dormand-Prince stages and returns `(y_5th_order,
263/// error_estimate)`.
264///
265/// `state` (`u^n`) is mutated in-place by boundary condition application
266/// at stage 1, same contract as every other solver in this crate (see
267/// [`evaluate_derivative`]).
268fn dopri45_stages(
269    domain: &Domain,
270    chain: &[&dyn ContextCalculator],
271    state: &mut ContextValue,
272    t: f64,
273    dt: f64,
274) -> Result<(DVector<f64>, DVector<f64>), OxiflowError> {
275    // Stage 1.
276    let k1_val = evaluate_derivative(domain, chain, state, t, dt)?;
277    let u_field = state.as_scalar_field()?.clone();
278    let k1 = k1_val.as_scalar_field()?.clone();
279
280    // Stage 2.
281    let y2 = u_field.clone() + k1.clone() * (dt * A21);
282    let mut s2 = ContextValue::ScalarField(y2);
283    let k2_val = evaluate_derivative(domain, chain, &mut s2, t + C2 * dt, dt)?;
284    let k2 = k2_val.as_scalar_field()?.clone();
285
286    // Stage 3.
287    let y3 = u_field.clone() + k1.clone() * (dt * A31) + k2.clone() * (dt * A32);
288    let mut s3 = ContextValue::ScalarField(y3);
289    let k3_val = evaluate_derivative(domain, chain, &mut s3, t + C3 * dt, dt)?;
290    let k3 = k3_val.as_scalar_field()?.clone();
291
292    // Stage 4.
293    let y4 = u_field.clone()
294        + k1.clone() * (dt * A41)
295        + k2.clone() * (dt * A42)
296        + k3.clone() * (dt * A43);
297    let mut s4 = ContextValue::ScalarField(y4);
298    let k4_val = evaluate_derivative(domain, chain, &mut s4, t + C4 * dt, dt)?;
299    let k4 = k4_val.as_scalar_field()?.clone();
300
301    // Stage 5.
302    let y5s = u_field.clone()
303        + k1.clone() * (dt * A51)
304        + k2.clone() * (dt * A52)
305        + k3.clone() * (dt * A53)
306        + k4.clone() * (dt * A54);
307    let mut s5 = ContextValue::ScalarField(y5s);
308    let k5_val = evaluate_derivative(domain, chain, &mut s5, t + C5 * dt, dt)?;
309    let k5 = k5_val.as_scalar_field()?.clone();
310
311    // Stage 6.
312    let y6 = u_field.clone()
313        + k1.clone() * (dt * A61)
314        + k2.clone() * (dt * A62)
315        + k3.clone() * (dt * A63)
316        + k4.clone() * (dt * A64)
317        + k5.clone() * (dt * A65);
318    let mut s6 = ContextValue::ScalarField(y6);
319    let k6_val = evaluate_derivative(domain, chain, &mut s6, t + C6 * dt, dt)?;
320    let k6 = k6_val.as_scalar_field()?.clone();
321
322    // Stage 7 -- A72 is 0.0, the k2 term is omitted entirely. The state
323    // this stage is evaluated at, `y7`, *is* the 5th-order solution
324    // (FSAL: b_i == a7i for i=1..6, b7 == 0 -- see module docs).
325    let y7 = u_field.clone()
326        + k1.clone() * (dt * A71)
327        + k3.clone() * (dt * A73)
328        + k4.clone() * (dt * A74)
329        + k5.clone() * (dt * A75)
330        + k6.clone() * (dt * A76);
331    let mut s7 = ContextValue::ScalarField(y7.clone());
332    let k7_val = evaluate_derivative(domain, chain, &mut s7, t + C7 * dt, dt)?;
333    let k7 = k7_val.as_scalar_field()?.clone();
334
335    let y_5th = y7;
336
337    // Error estimate: dt * sum((b_i - b*_i) * k_i).
338    let error: DVector<f64> = k1 * (dt * E1)
339        + k2 * (dt * E2)
340        + k3 * (dt * E3)
341        + k4 * (dt * E4)
342        + k5 * (dt * E5)
343        + k6 * (dt * E6)
344        + k7 * (dt * E7);
345
346    Ok((y_5th, error))
347}
