1//! # Module `solver::methods::bdf2`
2//!
3//! BDF2 integrator — implicit multi-step, 2nd order (issue #44).
4//!
5//! ## Algorithm
6//!
7//! $$\frac{3}{2}u^{n+1} - 2u^n + \frac{1}{2}u^{n-1} = \Delta t \cdot f(u^{n+1}, t^{n+1})$$
8//!
9//! Unlike Backward Euler / Crank-Nicolson ([`super::implicit`]), this needs
10//! **two** past states, not one — `SteppableSolver::history_depth()`
11//! returns `1` here (DD-034), and [`BDF2Solver::step`] receives `u^{n-1}`
12//! via the `history` parameter.
13//!
14//! `bdf2_step` (below) is BDF2-specific and stays in this file rather than
15//! [`super::implicit`]: unlike `theta_method_step`, which serves two
16//! solvers (Backward Euler, Crank-Nicolson), this formula has exactly one
17//! consumer. It still reuses [`super::implicit::finite_difference_jacobian`]
18//! and [`super::evaluate_derivative`] — the genuinely shared pieces.
19//!
20//! ## Startup (acceptance criterion, #44)
21//!
22//! BDF2 needs `u^{n-1}`, which doesn't exist at the very first step. When
23//! `history` is empty, [`BDF2Solver::step`] falls back to a single
24//! Backward Euler step (θ=1, via [`super::implicit::theta_method_step`])
25//! to produce `u^1` from `u^0` — the standard bootstrap for 2-step methods.
26//! From the second step onward, the real BDF2 update runs.
27//!
28//! ## Scope at J4a — fixed `dt` only
29//!
30//! *"Step-size change logic for adaptive dt"* (as #44 originally asked)
31//! is **not implemented**: `StepControl::Adaptive` doesn't exist yet
32//! anywhere in `oxiflow` (rejected by every solver, including this one),
33//! and #42 (DoPri45) — the issue that would introduce it — hasn't landed.
34//! Writing a variable-step BDF2 formula now would mean testing it against
35//! nothing real. Revisit once #42 lands; the classical fixed-step
36//! coefficients used here (3/2, -2, 1/2) would need to become
37//! ratio-dependent for a varying step size between `u^{n-1}` and `u^n`.
38//!
39//! ## Stability
40//!
41//! A-stable and, unlike Crank-Nicolson, L-stable — strong damping on
42//! stiff modes, same qualitative behaviour as Backward Euler but 2nd
43//! order accurate.
44

56/// BDF2 solver — implicit multi-step, 2nd order.
57///
58/// See [module docs](self) for the startup phase and the fixed-`dt`-only
59/// scope at J4a.
60pub struct BDF2Solver {
61    linear_solver: Box<dyn LinearSolver>,
62}
63
64impl Default for BDF2Solver {
65    fn default() -> Self {
66        Self {
67            linear_solver: Box::new(NalgebraDenseSolver),
68        }
69    }
70}

72impl BDF2Solver {
73    /// Creates a solver using the default `nalgebra` dense backend.
74    pub fn new() -> Self {
75        Self::default()
76    }

78    /// Substitutes the linear solver backend (DD-013).
79    pub fn with_linear_solver(mut self, linear_solver: Box<dyn LinearSolver>) -> Self {
80        self.linear_solver = linear_solver;
81        self
82    }

85impl Solver for BDF2Solver {
86    fn solve(
87        &self,
88        scenario: &Scenario,
89        config: &SolverConfiguration,
90    ) -> Result<SimulationResult, OxiflowError> {
91        // `solve_fixed_step` (DD-034 follow-up) manages the history buffer
92        // generically via `self.history_depth()` -- exactly equivalent to
93        // the hand-written `history = vec![u]` this used to do, since
94        // `insert(0, prev); truncate(1)` is the same operation for a
95        // depth-1 buffer. No BDF2-specific loop needed any more.
96        self.solve_fixed_step(scenario, config)
97    }
98}

100impl SteppableSolver for BDF2Solver {
101    fn history_depth(&self) -> usize {
102        1
103    }
104
105    fn step(
106        &self,
107        domain: &Domain,
108        chain: &[&dyn ContextCalculator],
109        state: &mut ContextValue,
110        history: &[ContextValue],
111        t: f64,
112        dt: f64,
113    ) -> Result<ContextValue, OxiflowError> {
114        match history.first() {
115            // Startup (#44 acceptance criterion): no u^{n-1} yet -- bootstrap
116            // with one Backward Euler step to produce u^1 from u^0.
117            None => theta_method_step(
118                domain,
119                chain,
120                state,
121                t,
122                dt,
123                1.0,
124                self.linear_solver.as_ref(),
125            ),
126            Some(u_prev) => bdf2_step(
127                domain,
128                chain,
129                state,
130                u_prev,
131                t,
132                dt,
133                self.linear_solver.as_ref(),
134            ),
135        }
136    }
137}

139/// Performs one BDF2 step, given `u^{n-1}` (`u_prev`) and `u^n` (`state`).
140///
141/// Same frozen-Jacobian, single-Newton-correction approach as
142/// [`super::implicit::theta_method_step`] (DD-033): exact when `f` is
143/// affine in `u`, a first-order approximation otherwise.
144///
145/// Derivation: linearising `g(u) = (3/2)u - 2u^n + (1/2)u^{n-1} - dt*f(u)`
146/// at `u = u^n` gives `g(u^n) = -(1/2)u^n + (1/2)u^{n-1} - dt*f(u^n)` and
147/// Jacobian `(3/2)I - dt*J_f`. The correction solves
148/// `[(3/2)I - dt*J_f] * delta_u = -g(u^n)`.
149fn bdf2_step(
150    domain: &Domain,
151    chain: &[&dyn ContextCalculator],
152    state: &mut ContextValue,
153    u_prev: &ContextValue,
154    t: f64,
155    dt: f64,
156    linear_solver: &dyn LinearSolver,
157) -> Result<ContextValue, OxiflowError> {
158    // f(u^n, t) -- BCs applied in-place to `state` itself.
159    let f_n = evaluate_derivative(domain, chain, state, t, dt)?;
160    let u_n_field = state.as_scalar_field()?.clone();
161    let f_n_field = f_n.as_scalar_field()?.clone();
162    let u_prev_field = u_prev.as_scalar_field()?.clone();
163
164    let n = u_n_field.len();
165    if u_prev_field.len() != n {
166        return Err(OxiflowError::InvalidDomain(format!(
167            "history state length {} != current state length {n}",
168            u_prev_field.len()
169        )));
170    }
171
172    // Jacobian frozen at u^n, evaluated at the target time t + dt.
173    let jacobian = finite_difference_jacobian(domain, chain, state, t + dt, dt)?;
174
175    // System matrix: (3/2)*I - dt*J_f.
176    let identity = DMatrix::<f64>::identity(n, n);
177    let system_matrix = identity * 1.5 - jacobian * dt;
178
179    // RHS: -g(u^n) = (1/2)*u^n - (1/2)*u^{n-1} + dt*f(u^n). Built from
180    // fully-owned operands throughout -- see DD-033 rationale on avoiding
181    // ambiguous reference/owned nalgebra operator resolution.
182    let half_u_n = u_n_field.clone() * 0.5;
183    let half_u_prev = u_prev_field * 0.5;
184    let dt_f_n = f_n_field * dt;
185    let rhs: DVector<f64> = half_u_n + dt_f_n - half_u_prev;
186
187    let delta_u = linear_solver.solve(&system_matrix, &rhs)?;
188
189    let u_next: DVector<f64> = u_n_field + delta_u;
190    Ok(ContextValue::ScalarField(u_next))
191}

383    #[test]
384    fn exponential_decay_matches_recurrence_after_startup() {
385        // Independent reimplementation of the BDF2 recurrence (derived by
386        // hand, see module docs) -- cross-checks the solver's output
387        // rather than re-deriving the same formula from its own code.
388        let lambda = 2.0;
389        let dt = 0.1;
390        let n_steps: usize = 10;
391        let scenario = Scenario::single(Box::new(ExponentialDecay { lambda }), make_mesh(2));
392        let config = make_config(n_steps as f64 * dt, dt);
393        let result = BDF2Solver::new().solve(&scenario, &config).unwrap();
394
395        let mut u_prev = 1.0_f64; // u^0
396        let mut u_curr = u_prev / (1.0 + lambda * dt); // u^1, Backward Euler startup
397        for _ in 1..n_steps {
398            let u_next = (2.0 * u_curr - 0.5 * u_prev) / (1.5 + lambda * dt);
399            u_prev = u_curr;
400            u_curr = u_next;
401        }
402
403        let final_field = result.states.last().unwrap().as_scalar_field().unwrap();
404        for v in final_field.iter() {
405            assert!((v - u_curr).abs() < 1e-9, "got {v}, expected {u_curr}");
406        }
407    }
