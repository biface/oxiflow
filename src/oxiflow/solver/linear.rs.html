1//! # Module `solver::linear`
2//!
3//! Linear system solver abstraction — DD-013.
4//!
5//! Lives at `solver::linear`, not nested under `solver::methods`: solving
6//! `A * x = b` knows nothing about `t`, `dt`, or any `Domain` — it is a
7//! generic service, not a temporal-integration concern. The implicit
8//! integrators ([`crate::solver::methods::implicit`]) are its first
9//! consumer, but DD-013 already anticipates a second one unrelated to time
10//! integration: sparse FEM systems at v2.0/J7, via a `faer` backend
11//! implementing this same trait.
12//!
13//! [`NalgebraDenseSolver`] (dense LU) is the default for the small systems
14//! implicit integrators produce at J4a. The sparse `faer` backend arrives
15//! at v0.5.0 (DD-013, second phase) as an additional implementation of
16//! this same trait — no change needed here when it does.
17
18use nalgebra::{DMatrix, DVector};
19
20use crate::context::error::OxiflowError;
21
22/// Solves a dense linear system `A * x = b`.
23///
24/// Implementors may assume `a` is square and `b.len() == a.nrows()` —
25/// callers are responsible for that invariant (see
26/// [`crate::solver::methods::implicit`]).
27pub trait LinearSolver: Send + Sync {
28    /// Solves `a * x = b` for `x`.
29    ///
30    /// # Errors
31    ///
32    /// Returns `OxiflowError::PreconditionFailed` if the system is singular
33    /// or near-singular (decomposition failure).
34    fn solve(&self, a: &DMatrix<f64>, b: &DVector<f64>) -> Result<DVector<f64>, OxiflowError>;
35}
36
37/// Default backend — dense LU decomposition via `nalgebra`.
38///
39/// Appropriate for the small, dense systems implicit integrators (Backward
40/// Euler, Crank-Nicolson) produce before `DiscreteOperator` (INV-2,
41/// v0.5.0+) introduces large sparse operators.
42#[derive(Debug, Default, Clone, Copy)]
43pub struct NalgebraDenseSolver;
44
45impl LinearSolver for NalgebraDenseSolver {
46    fn solve(&self, a: &DMatrix<f64>, b: &DVector<f64>) -> Result<DVector<f64>, OxiflowError> {
47        a.clone()
48            .lu()
49            .solve(b)
50            .ok_or_else(|| OxiflowError::PreconditionFailed {
51                context: "NalgebraDenseSolver",
52                message: "linear system is singular or near-singular (LU decomposition failed)"
53                    .into(),
54            })
55    }
56}

58// ── Tests ─────────────────────────────────────────────────────────────────────
59
60#[cfg(test)]
61mod tests {
62    use super::*;
63
64    #[test]
65    fn solves_identity_system() {
66        let a = DMatrix::<f64>::identity(3, 3);
67        let b = DVector::from_vec(vec![1.0, 2.0, 3.0]);
68        let x = NalgebraDenseSolver.solve(&a, &b).unwrap();
69        assert!((x[0] - 1.0).abs() < 1e-12);
70        assert!((x[1] - 2.0).abs() < 1e-12);
71        assert!((x[2] - 3.0).abs() < 1e-12);
72    }
73
74    #[test]
75    fn solves_diagonal_system() {
76        // 2x = 4, 5y = 10 -> x=2, y=2
77        let a = DMatrix::from_diagonal(&DVector::from_vec(vec![2.0, 5.0]));
78        let b = DVector::from_vec(vec![4.0, 10.0]);
79        let x = NalgebraDenseSolver.solve(&a, &b).unwrap();
80        assert!((x[0] - 2.0).abs() < 1e-12);
81        assert!((x[1] - 2.0).abs() < 1e-12);
82    }
83
84    #[test]
85    fn singular_system_returns_error() {
86        // Rank-deficient: second row is a multiple of the first.
87        let a = DMatrix::from_row_slice(2, 2, &[1.0, 1.0, 2.0, 2.0]);
88        let b = DVector::from_vec(vec![1.0, 2.0]);
89        let err = NalgebraDenseSolver.solve(&a, &b).unwrap_err();
90        assert!(matches!(err, OxiflowError::PreconditionFailed { .. }));
91    }
92}
