1//! # Module `solver`
2//!
3//! Numerical solving orchestration — WHAT/HOW separation (issue #32).
4//!
5//! ## Responsibilities
6//!
7//! | Type | Role |
8//! |---|---|
9//! | [`scenario::Scenario`] | Declares the problem (WHAT) |
10//! | [`config::SolverConfiguration`] | Configures solving (HOW) |
11//! | [`Solver`] | Orchestrates execution |
12//!
13//! ## Contractual execution order
14//!
15//! `Solver::solve()` implementations must follow this order at each time step:
16//!
17//! 1. **Calculators** — populate `ComputeContext` in topological order
18//! 2. **Boundary conditions** — apply to state using `ctx` (J2)
19//! 3. **`compute_physics`** — compute `du/dt` from state + context
20//! 4. **Integrate** — advance state by `dt`
21//!
22//! This order is a contract, not a convention. Deviating from it produces
23//! silently incorrect results.

35// ── SimulationResult ──────────────────────────────────────────────────────────
36
37/// Result of a completed simulation.
38///
39/// `states` and `times` have the same length. The save frequency is controlled
40/// by `SolverConfiguration::time.save_every`.
41///
42/// # Examples
43///
44/// ```rust, ignore
45/// use oxiflow::solver::SimulationResult;
46/// use oxiflow::context::value::ContextValue;
47/// use nalgebra::DVector;
48///
49/// let result = SimulationResult {
50///     states: vec![ContextValue::ScalarField(DVector::from_element(10, 0.0))],
51///     times:  vec![0.0],
52///     n_steps: 1,
53/// };
54/// assert_eq!(result.states.len(), result.times.len());
55/// ```
56#[non_exhaustive]
57#[derive(Debug)]
58pub struct SimulationResult {
59    /// Saved field states at each recorded time.
60    pub states: Vec<crate::context::value::ContextValue>,
61    /// Simulation times corresponding to each saved state.
62    pub times: Vec<f64>,
63    /// Total number of time steps taken (may be larger than `states.len()`
64    /// if `save_every > 1`).
65    pub n_steps: usize,
66    /// Solver metadata: timing, rejected steps, convergence info.
67    ///
68    /// Keys follow the convention `"solver.<key>"` (e.g. `"solver.rejected_steps"`).
69    /// Empty at J1 — populated by adaptive integrators at J4 (DoPri45, BDF2).
70    pub metadata: std::collections::HashMap<String, f64>,
71}
72
73impl SimulationResult {
74    /// Returns the number of saved states.
75    pub fn len(&self) -> usize {
76        self.states.len()
77    }
78
79    /// Returns `true` if no states were saved.
80    pub fn is_empty(&self) -> bool {
81        self.states.is_empty()
82    }
83
84    /// Returns the final simulation time.
85    pub fn t_final(&self) -> Option<f64> {
86        self.times.last().copied()
87    }
88}

90// ── Solver trait ──────────────────────────────────────────────────────────────
91
92/// Orchestrates the time integration loop.
93///
94/// Implementations receive a `Scenario` (WHAT) and a `SolverConfiguration`
95/// (HOW) and execute the contractual loop until `t_end`.
96///
97/// At J1, `Solver` implementations must:
98/// - Verify `scenario.n_domains() == 1` (multi-domain requires J3)
99/// - Build the calculator chain via `chain::build_calculator_chain()`
100/// - Follow the contractual execution order
101///
102/// # Object safety
103///
104/// This trait is object-safe to support INV-4 (plugin-safe API, v2.0).
105pub trait Solver: Send + Sync {
106    /// Runs the simulation and returns the collected states.
107    fn solve(
108        &self,
109        scenario: &Scenario,
110        config: &SolverConfiguration,
111    ) -> Result<SimulationResult, OxiflowError>;
112}

116#[cfg(test)]
117mod tests {
118    use super::*;
119    use crate::context::value::ContextValue;
120    use nalgebra::DVector;
121
122    #[test]
123    fn simulation_result_len() {
124        let r = SimulationResult {
125            states: vec![ContextValue::ScalarField(DVector::from_element(5, 0.0))],
126            times: vec![1.0],
127            n_steps: 10,
128            metadata: std::collections::HashMap::new(),
129        };
130        assert_eq!(r.len(), 1);
131        assert!(!r.is_empty());
132    }
133
134    #[test]
135    fn simulation_result_t_final() {
136        let r = SimulationResult {
137            states: vec![
138                ContextValue::ScalarField(DVector::from_element(5, 0.0)),
139                ContextValue::ScalarField(DVector::from_element(5, 1.0)),
140            ],
141            times: vec![0.0, 1.0],
142            n_steps: 2,
143            metadata: std::collections::HashMap::new(),
144        };
145        assert_eq!(r.t_final(), Some(1.0));
146    }
147
148    #[test]
149    fn empty_result() {
150        let r = SimulationResult {
151            states: vec![],
152            times: vec![],
153            n_steps: 0,
154            metadata: std::collections::HashMap::new(),
155        };
156        assert!(r.is_empty());
157        assert_eq!(r.t_final(), None);
158    }
159
160    #[test]
161    fn solver_is_object_safe() {
162        fn assert_object_safe<T: Solver + ?Sized>() {}
163        assert_object_safe::<dyn Solver>();
164    }
165}
