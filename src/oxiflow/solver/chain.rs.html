1//! # Module `solver::chain`
2//!
3//! Calculator chain — validation and ordered execution (issues #33, #36, DD-009).
4//!
5//! ## Responsibility
6//!
7//! `build_calculator_chain` verifies that every context variable required by the
8//! scenario has a corresponding calculator in `SolverConfiguration`, then returns
9//! the calculators in a correct execution order.
10//!
11//! ## Ordering strategy (DD-009)
12//!
13//! A hybrid two-path algorithm is selected at chain-build time:
14//!
15//! **Priority path (fast path)** — no calculator declares a non-empty `depends_on()`.
16//! Calculators are sorted by ascending `priority()` (stable sort, O(n log n)).
17//! A total order by priority cannot contain a cycle. Priority ranges:
18//! - **0–49**: system variables (`Time`, `TimeStep`) — injected directly by the solver
19//! - **50–99**: external data providers
20//! - **100+**: derived quantities (default)
21//!
22//! **Topological path (Kahn)** — at least one calculator declares a non-empty
23//! `depends_on()`. A directed acyclic graph is built from the declared dependencies:
24//! if calculator C declares `depends_on: [X]`, an edge is drawn from every calculator
25//! providing X to C. Within each topological tier, calculators are ordered by
26//! ascending `priority()`. A cycle returns
27//! [`OxiflowError::CircularDependency`].
28//!
29//! ## Built-in variables
30//!
31//! `Time` and `TimeStep` are always available in `ComputeContext` — the solver
32//! injects them directly via `ComputeContext::new(t, dt)` before running the
33//! chain. Built-in variables declared in `depends_on()` are ignored when building
34//! the dependency graph.
35

40/// Built-in variables provided directly by the solver — no calculator needed.
41///
42/// `Time` and `TimeStep` are injected via `ComputeContext::new(t, dt)` before
43/// the calculator chain runs. Declaring them as requirements is valid; checking
44/// them against the calculator list would be a false negative. Declaring them in
45/// `depends_on()` is valid and ignored when building the dependency graph.
46const BUILTIN_VARIABLES: &[ContextVariable] = &[ContextVariable::Time, ContextVariable::TimeStep];
47
48/// Validates requirements against provided calculators and returns an
49/// execution-ordered chain.
50///
51/// # Validation
52///
53/// Every variable in `requirements` must be either:
54/// - a built-in variable (`Time`, `TimeStep`), or
55/// - covered by at least one calculator via `provides()`.
56///
57/// A variable covered by multiple calculators is accepted — the last one
58/// to execute wins (consistent with `ComputeContext::insert` overwrite).
59///
60/// # Ordering (DD-009)
61///
62/// If no calculator declares a non-empty `depends_on()`, calculators are sorted
63/// by ascending `priority()` (stable sort — registration order preserved within
64/// equal priorities).
65///
66/// If at least one calculator declares a non-empty `depends_on()`, Kahn's
67/// algorithm is applied on the full dependency graph. Within each topological
68/// tier, calculators are sorted by ascending `priority()`. Built-in variables
69/// in `depends_on()` declarations are ignored when building the graph.
70///
71/// # Errors
72///
73/// - [`OxiflowError::MissingCalculator`] — first uncovered non-builtin requirement.
74/// - [`OxiflowError::CircularDependency`] — cycle detected in the dependency graph
75///   (Kahn path only).
76///
77/// # Examples
78///
79/// ```rust
80/// use oxiflow::solver::chain::build_calculator_chain;
81/// use oxiflow::context::variable::ContextVariable;
82/// use oxiflow::context::value::ContextValue;
83/// use oxiflow::context::compute::ComputeContext;
84/// use oxiflow::context::error::OxiflowError;
85/// use oxiflow::context::calculator::ContextCalculator;
86/// use oxiflow::model::traits::RequiresContext;
87///
88/// #[derive(Debug)]
89/// struct TimeCalc;
90/// impl RequiresContext for TimeCalc {
91///     fn required_variables(&self) -> Vec<ContextVariable> { vec![] }
92///     fn priority(&self) -> u32 { 0 }
93/// }
94/// impl ContextCalculator for TimeCalc {
95///     fn provides(&self) -> ContextVariable { ContextVariable::Time }
96///     fn compute(&self, _: &ContextValue, ctx: &ComputeContext)
97///         -> Result<ContextValue, OxiflowError>
98///     { Ok(ContextValue::Scalar(ctx.time())) }
99/// }
100///
101/// let requirements = vec![ContextVariable::Time];
102/// let calculators: Vec<Box<dyn ContextCalculator>> = vec![Box::new(TimeCalc)];
103/// let chain = build_calculator_chain(&requirements, &calculators).unwrap();
104/// assert_eq!(chain.len(), 1);
105/// ```
106pub fn build_calculator_chain<'a>(
107    requirements: &[ContextVariable],
108    calculators: &'a [Box<dyn ContextCalculator>],
109) -> Result<Vec<&'a dyn ContextCalculator>, OxiflowError> {
110    // Validate: every non-builtin requirement must have a provider.
111    for req in requirements {
112        if is_builtin(req) {
113            continue;
114        }
115        let covered = calculators.iter().any(|c| &c.provides() == req);
116        if !covered {
117            return Err(OxiflowError::MissingCalculator(req.clone()));
118        }
119    }
120
121    // Select ordering path based on whether any calculator declares dependencies.
122    let has_deps = calculators.iter().any(|c| !c.depends_on().is_empty());
123    if has_deps {
124        build_kahn_chain(calculators)
125    } else {
126        build_priority_chain(calculators)
127    }
128}
129
130/// Fast path — no `depends_on()` declared. Stable sort by ascending priority.
131fn build_priority_chain(
132    calculators: &[Box<dyn ContextCalculator>],
133) -> Result<Vec<&dyn ContextCalculator>, OxiflowError> {
134    let mut chain: Vec<&dyn ContextCalculator> = calculators.iter().map(|c| c.as_ref()).collect();
135    chain.sort_by_key(|c| c.priority());
136    Ok(chain)
137}
138
139/// Kahn path — topological sort with priority as tiebreaker within each tier.
140///
141/// Edges: for each calculator C declaring `depends_on: [X]`, draw an edge from
142/// every calculator providing X to C. Built-in variables are excluded from the
143/// graph. A cycle returns `CircularDependency` naming the first `depends_on`
144/// variable of the first blocked calculator found.
145fn build_kahn_chain(
146    calculators: &[Box<dyn ContextCalculator>],
147) -> Result<Vec<&dyn ContextCalculator>, OxiflowError> {
148    let n = calculators.len();
149
150    // Build adjacency list and in-degree table.
151    // successors[j] = indices of calculators that must run after j.
152    let mut successors: Vec<Vec<usize>> = vec![vec![]; n];
153    let mut in_degree: Vec<usize> = vec![0; n];
154
155    for (i, calc) in calculators.iter().enumerate() {
156        for dep_var in calc.depends_on() {
157            if is_builtin(&dep_var) {
158                continue;
159            }
160            for (j, provider) in calculators.iter().enumerate() {
161                if provider.provides() == dep_var {
162                    successors[j].push(i);
163                    in_degree[i] += 1;
164                }
165            }
166        }
167    }
168
169    // Initial queue: calculators with no unresolved predecessors, by priority.
170    let mut queue: Vec<usize> = (0..n).filter(|&i| in_degree[i] == 0).collect();
171    queue.sort_by_key(|&i| calculators[i].priority());
172
173    let mut result: Vec<&dyn ContextCalculator> = Vec::with_capacity(n);
174
175    while !queue.is_empty() {
176        // Emit the first element (lowest priority in current tier).
177        let i = queue.remove(0);
178        result.push(calculators[i].as_ref());
179
180        // Decrement in-degree of successors; collect newly unblocked nodes.
181        let mut newly_free: Vec<usize> = Vec::new();
182        for &j in &successors[i] {
183            in_degree[j] -= 1;
184            if in_degree[j] == 0 {
185                newly_free.push(j);
186            }
187        }
188
189        // Merge newly unblocked nodes into the queue, maintaining priority order.
190        queue.extend(newly_free);
191        queue.sort_by_key(|&i| calculators[i].priority());
192    }
193
194    // If not all calculators were emitted, a cycle exists.
195    if result.len() < n {
196        let blocked = (0..n).find(|&i| in_degree[i] > 0).unwrap();
197        let var = calculators[blocked]
198            .depends_on()
199            .into_iter()
200            .find(|v| !is_builtin(v))
201            .unwrap_or_else(|| calculators[blocked].provides());
202        return Err(OxiflowError::CircularDependency(var));
203    }
204
205    Ok(result)
206}
207
208fn is_builtin(var: &ContextVariable) -> bool {
209    BUILTIN_VARIABLES.contains(var)
210}

397        assert_eq!(chain[3].provides(), var("W"));

399        let middle: Vec<ContextVariable> = chain[1..3].iter().map(|c| c.provides()).collect();
400        assert!(middle.contains(&var("Y")));

420    fn kahn_multiple_providers_all_precede_dependent() {
421        // A and B both provide X; C depends on X.
422        // Both A and B must appear before C.
423        let calcs = vec![
424            make_deps_calc(var("Z"), 100, vec![var("X")]), // C
425            make_calc(var("X"), 60),                       // B
426            make_calc(var("X"), 50),                       // A (lower priority)
427        ];
428        let chain = build_calculator_chain(&[], &calcs).unwrap();
429        let c_pos = chain.iter().position(|c| c.provides() == var("Z")).unwrap();
430        let a_pos = chain
431            .iter()
432            .position(|c| c.priority() == 50 && c.provides() == var("X"))
433            .unwrap();
434        let b_pos = chain
435            .iter()
436            .position(|c| c.priority() == 60 && c.provides() == var("X"))
437            .unwrap();
438        assert!(a_pos < c_pos);
439        assert!(b_pos < c_pos);
440    }

479    fn kahn_mixed_with_and_without_deps() {
480        // Calculators without depends_on coexist with one that has deps.
481        // Independent ones (no deps) run by priority; the dependent one after its provider.
482        let calcs = vec![

485            make_calc(var("gamma"), 50),

488        // gamma (priority 50) and alpha (priority 200) have no deps on each other.

491            .iter()
492            .position(|c| c.provides() == var("alpha"))
493            .unwrap();
494        let beta_pos = chain
495            .iter()
496            .position(|c| c.provides() == var("beta"))
497            .unwrap();
498        assert!(alpha_pos < beta_pos);
499        // gamma has no deps → should appear first (priority 50 < 200).
500        assert_eq!(chain[0].provides(), var("gamma"));
