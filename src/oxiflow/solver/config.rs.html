1//! # Module `solver::config`
2//!
3//! Solving configuration — `SolverConfiguration` (HOW pole, DD-021, issue #32).
4//!
5//! ## Design
6//!
7//! `dt: f64` is never exposed directly in the public API. Instead, `StepControl`
8//! encapsulates both fixed-step and adaptive-step strategies. `TimeConfiguration`
9//! groups all temporal parameters. This prevents the breaking change that would
10//! occur at J4 when adaptive integrators (DoPri45, BDF2) are introduced (DD-021).
11
12use crate::context::calculator::ContextCalculator;
13
14// ── StepControl ───────────────────────────────────────────────────────────────
15
16/// Time step control strategy.
17///
18/// At J1, only `Fixed` is used. `Adaptive` is reserved for J4 (DoPri45, BDF2)
19/// and added as a new variant — non-breaking for all J1/J2 code.
20///
21/// # Examples
22///
23/// ```rust
24/// use oxiflow::solver::config::StepControl;
25///
26/// let fixed = StepControl::Fixed { dt: 0.01 };
27/// assert_eq!(fixed.dt_initial(), 0.01);
28/// ```
29#[non_exhaustive]
30#[derive(Debug, Clone)]
31#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
32pub enum StepControl {
33    /// Fixed time step — J1.
34    ///
35    /// Stability for explicit methods requires the CFL condition:
36    ///
37    /// $$\text{CFL} = \frac{v \, \Delta t}{\Delta x} \leq 1$$
38    Fixed {
39        /// Time step size.
40        dt: f64,
41    },
42
43    /// Adaptive step-size control — RESERVED J4 (DoPri45, BDF2, DD-021).
44    ///
45    /// The integrator adjusts $\Delta t$ to keep the local truncation error
46    /// within the tolerance band:
47    ///
48    /// $$\| e \| \leq \text{atol} + \text{rtol} \cdot \| u \|$$
49    Adaptive {
50        /// Initial time step guess.
51        dt_init: f64,
52        /// Minimum allowed time step — `OxiflowError::SolverDivergence` if reached.
53        dt_min: f64,
54        /// Maximum allowed time step.
55        dt_max: f64,
56        /// Relative tolerance.
57        rtol: f64,
58        /// Absolute tolerance.
59        atol: f64,
60    },
61}
62
63impl StepControl {
64    /// Returns the initial `dt` regardless of strategy.
65    pub fn dt_initial(&self) -> f64 {
66        match self {
67            Self::Fixed { dt } => *dt,
68            Self::Adaptive { dt_init, .. } => *dt_init,
69        }
70    }
71
72    /// Returns `true` if this is a fixed step strategy.
73    pub fn is_fixed(&self) -> bool {
74        matches!(self, Self::Fixed { .. })
75    }
76
77    /// Returns `true` if this is an adaptive step strategy.
78    pub fn is_adaptive(&self) -> bool {
79        matches!(self, Self::Adaptive { .. })
80    }
81}

83// ── IntegratorKind ────────────────────────────────────────────────────────────
84
85/// Temporal integration method.
86///
87/// `Euler` is active since J1; `RK4`, `BackwardEuler`, `CrankNicolson`,
88/// `BDF2`, `DoPri45`, `Imex` since J4a (#41, #43, #44, #42, #45).
89///
90/// # Examples
91///
92/// ```rust
93/// use oxiflow::solver::config::IntegratorKind;
94///
95/// let method = IntegratorKind::Euler;
96/// assert!(method.is_explicit());
97/// ```
98#[non_exhaustive]
99#[derive(Debug, Clone, PartialEq, Eq)]
100#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
101pub enum IntegratorKind {
102    /// Forward Euler — explicit, 1st order — J1.
103    Euler,
104    /// Runge-Kutta 4 — explicit, 4th order — J4a (#41).
105    RK4,
106    /// Backward Euler — implicit, 1st order — J4a (#43, DD-013, DD-033).
107    BackwardEuler,
108    /// Crank-Nicolson — semi-implicit, 2nd order — J4a (#43, DD-013, DD-033).
109    CrankNicolson,
110    /// BDF2 — implicit multi-step, 2nd order — J4a (#44, DD-034). Fixed
111    /// `dt` only; see `BDF2Solver` module docs for the deferred
112    /// variable-step formula.
113    BDF2,
114    /// Dormand-Prince DoPri45 — explicit, adaptive step, order 5 — J4a
115    /// (#42, DD-036). `Solver` only, not `SteppableSolver` — see
116    /// `DoPri45Solver` module docs.
117    DoPri45,
118    /// Operator splitting (Strang) — n ≥ 2 sub-models, each with its own
119    /// integrator — J4a (#45, DD-037). `Solver` only, not
120    /// `SteppableSolver` — see `OperatorSplittingSolver` module docs.
121    /// Neither purely explicit nor purely implicit — excluded from
122    /// `is_explicit()`.
123    Imex,
124}
125
126impl IntegratorKind {
127    /// Returns `true` if the method is explicit.
128    pub fn is_explicit(&self) -> bool {
129        matches!(self, Self::Euler | Self::RK4 | Self::DoPri45)
130    }
131}

133// ── TimeConfiguration ─────────────────────────────────────────────────────────
134
135/// Temporal simulation parameters.
136///
137/// Groups `t_end`, step control strategy, and output frequency.
138/// Decoupled from `SolverConfiguration` so that time parameters can be
139/// modified independently of the integration method and calculators.
140///
141/// # Examples
142///
143/// ```rust
144/// use oxiflow::solver::config::{TimeConfiguration, StepControl};
145///
146/// let time = TimeConfiguration::new(600.0, StepControl::Fixed { dt: 0.1 });
147/// assert_eq!(time.t_end, 600.0);
148/// assert_eq!(time.n_steps_estimate(), 6000);
149/// ```
150#[non_exhaustive]
151#[derive(Debug, Clone)]
152#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
153pub struct TimeConfiguration {
154    /// End time of the simulation.
155    pub t_end: f64,
156    /// Step control strategy.
157    pub step_control: StepControl,
158    /// Save state every N steps into `SimulationResult`.
159    ///
160    /// `None` — save every step (default; suitable for short simulations).
161    /// `Some(n)` — save every n-th step (avoids large result vectors).
162    pub save_every: Option<usize>,
163}
164
165impl TimeConfiguration {
166    /// Creates a time configuration with default save frequency (every step).
167    pub fn new(t_end: f64, step_control: StepControl) -> Self {
168        Self {
169            t_end,
170            step_control,
171            save_every: None,
172        }
173    }
174
175    /// Sets the save frequency.
176    pub fn saving_every(mut self, n: usize) -> Self {
177        self.save_every = Some(n);
178        self
179    }
180
181    /// Estimates the number of steps for fixed step control.
182    ///
183    /// Returns 0 for adaptive step control (unknown a priori).
184    pub fn n_steps_estimate(&self) -> usize {
185        match &self.step_control {
186            StepControl::Fixed { dt } => {
187                if *dt > 0.0 {
188                    (self.t_end / dt).ceil() as usize
189                } else {
190                    0
191                }
192            }
193            StepControl::Adaptive { .. } => 0,
194        }
195    }
196}

198// ── SolverConfiguration ───────────────────────────────────────────────────────
199
200/// Solving configuration — HOW pole.
201///
202/// Groups the integration method, temporal parameters, and context calculators.
203/// `DiscreteOperator` (INV-2, J4b) is **not** a configuration field — it is
204/// an implementation detail inside spatial `ContextCalculator`s.
205///
206/// # Serialisation
207///
208/// `SolverConfiguration` does not implement `serde::Serialize` / `serde::Deserialize`.
209/// The `calculators` field holds `Vec<Box<dyn ContextCalculator>>` (trait objects),
210/// which cannot be serialised directly. The serialisable subset of configuration
211/// is captured in `SimulationSnapshot` (DD-025 Option B, v0.6.0).
212///
213/// # Examples
214///
215/// ```rust
216/// use oxiflow::solver::config::{
217///     SolverConfiguration, TimeConfiguration, StepControl, IntegratorKind,
218/// };
219///
220/// let config = SolverConfiguration::new(
221///     TimeConfiguration::new(600.0, StepControl::Fixed { dt: 0.1 }),
222///     IntegratorKind::Euler,
223/// );
224/// assert!(config.calculators.is_empty());
225/// assert_eq!(config.time.t_end, 600.0);
226/// ```
227#[non_exhaustive]
228pub struct SolverConfiguration {
229    /// Temporal parameters — t_end, step control, save frequency.
230    pub time: TimeConfiguration,
231    /// Temporal integration method.
232    pub integrator: IntegratorKind,
233    /// Context variable calculators provided by the user.
234    ///
235    /// The solver chains these in topological order to populate `ComputeContext`
236    /// at each time step. Built-in calculators (Time, TimeStep) are always added
237    /// automatically; only derived quantities need user-supplied calculators.
238    pub calculators: Vec<Box<dyn ContextCalculator>>,
239    // external_data: Option<Arc<dyn ExternalDataProvider>>  — RESERVED J2
240    // parallel_threshold: Option<usize>                     — RESERVED J5 (DD-014)
241}
242
243impl SolverConfiguration {
244    /// Creates a new solver configuration with no user calculators.
245    pub fn new(time: TimeConfiguration, integrator: IntegratorKind) -> Self {
246        Self {
247            time,
248            integrator,
249            calculators: Vec::new(),
250        }
251    }
252
253    /// Adds a context calculator (builder pattern).
254    pub fn with_calculator(mut self, calc: Box<dyn ContextCalculator>) -> Self {
255        self.calculators.push(calc);
256        self
257    }
258
259    /// Adds multiple context calculators at once.
260    pub fn with_calculators(mut self, calcs: Vec<Box<dyn ContextCalculator>>) -> Self {
261        self.calculators.extend(calcs);
262        self
263    }
264}

388        #[derive(Debug)]
389        struct DummyCalc;
390        impl RequiresContext for DummyCalc {
391            fn required_variables(&self) -> Vec<ContextVariable> {
392                vec![]
393            }
394        }
395        impl crate::context::calculator::ContextCalculator for DummyCalc {
396            fn provides(&self) -> ContextVariable {
397                ContextVariable::Time
398            }
399            fn compute(
400                &self,
401                _: &ContextValue,
402                ctx: &ComputeContext,
403            ) -> Result<ContextValue, OxiflowError> {
404                Ok(ContextValue::Scalar(ctx.time()))
405            }
406        }
407
408        let cfg = SolverConfiguration::new(
409            TimeConfiguration::new(1.0, StepControl::Fixed { dt: 0.1 }),
410            IntegratorKind::RK4,
411        )
412        .with_calculator(Box::new(DummyCalc));
413
414        assert_eq!(cfg.calculators.len(), 1);
415        assert_eq!(cfg.integrator, IntegratorKind::RK4);
416    }
