1//! # Module `solver::orchestrator`
2//!
3//! Multi-domain orchestration — DD-031.
4//!
5//! ## Why this exists
6//!
7//! [`Solver`](crate::solver::Solver) is deliberately single-domain — it
8//! drives one [`Domain`](crate::solver::scenario::Domain) through a full
9//! time range. Coupled scenarios
10//! (lahar–lake, #40) need several domains advancing together, each
11//! exchanging state with the others via
12//! [`CouplingOperator`](crate::coupling::CouplingOperator) (INV-3,
13//! DD-011) between steps.
14//!
15//! Rather than generalising `Solver`/`SimulationResult` for this — which
16//! would force every coupled domain through the same integrator —
17//! [`MultiDomainOrchestrator`] drives one [`SteppableSolver`] *per domain*,
18//! so the lahar domain can run `ForwardEulerSolver` while the lake domain
19//! runs `RK4Solver`, or any other combination.
20//!
21//! ## Scope (v1)
22//!
23//! `dt` is synchronised across all domains: every domain advances by the
24//! same step before couplings are applied. Per-domain `dt` (multirate /
25//! sub-cycling) is a substantially harder problem — interface time
26//! interpolation, coupling stability — deliberately deferred until a
27//! concrete case requires it (see DD-031).
28//!
29//! ## Per-step order
30//!
31//! At each synchronised step:
32//!
33//! 1. **Advance every domain** by one step, each via its own registered
34//!    `SteppableSolver` (which itself follows the contractual
35//!    calculators → boundary conditions → `compute_physics` order, see
36//!    [`crate::solver`]).
37//! 2. **Apply every registered `CouplingOperator`**, in declaration order,
38//!    each reading the just-updated [`MultiDomainState`] and returning an
39//!    updated one.
40//! 3. **Guard against divergence** across all domains.
41

54// ── OrchestratorConfig ──────────────────────────────────────────────────────────
55
56/// Configuration shared by every domain in an orchestrated run.
57///
58/// Mirrors [`SolverConfiguration`](crate::solver::config::SolverConfiguration)
59/// minus `integrator` — integrator choice is per-domain here, via
60/// [`MultiDomainOrchestrator::with_domain`].
61#[non_exhaustive]
62#[derive(Debug)]
63pub struct OrchestratorConfig {
64    /// Temporal parameters — t_end, step control, save frequency. Only
65    /// `StepControl::Fixed` is supported (DD-031 v1: synchronised `dt`).
66    pub time: TimeConfiguration,
67    /// Context variable calculators, shared across all domains — built
68    /// once from `scenario.context_requirements()`, which already
69    /// aggregates every domain's needs.
70    pub calculators: Vec<Box<dyn ContextCalculator>>,
71}
72
73impl OrchestratorConfig {
74    /// Creates a new configuration with no calculators.
75    pub fn new(time: TimeConfiguration) -> Self {
76        Self {
77            time,
78            calculators: Vec::new(),
79        }
80    }
81
82    /// Adds a context calculator (builder pattern).
83    pub fn with_calculator(mut self, calc: Box<dyn ContextCalculator>) -> Self {
84        self.calculators.push(calc);
85        self
86    }
87}

89// ── MultiDomainSimulationResult ──────────────────────────────────────────────────
90
91/// Result of a completed multi-domain orchestrated run.
92///
93/// Distinct from [`SimulationResult`](crate::solver::SimulationResult) —
94/// one entry per saved time, each holding every domain's state rather than
95/// a single domain's `ContextValue`.
96#[non_exhaustive]
97#[derive(Debug, Clone)]
98pub struct MultiDomainSimulationResult {
99    /// Saved multi-domain states at each recorded time.
100    pub states: Vec<MultiDomainState>,
101    /// Simulation times corresponding to each saved state.
102    pub times: Vec<f64>,
103    /// Total number of synchronised steps taken (may be larger than
104    /// `states.len()` if `save_every > 1`).
105    pub n_steps: usize,
106}
107
108impl MultiDomainSimulationResult {
109    /// Returns the number of saved states.
110    pub fn len(&self) -> usize {
111        self.states.len()
112    }
113
114    /// Returns `true` if no states were saved.
115    pub fn is_empty(&self) -> bool {
116        self.states.is_empty()
117    }
118
119    /// Returns the final simulation time.
120    pub fn t_final(&self) -> Option<f64> {
121        self.times.last().copied()
122    }
123}

125// ── MultiDomainOrchestrator ───────────────────────────────────────────────────────
126
127/// Drives multiple coupled [`Domain`](crate::solver::scenario::Domain)s,
128/// each with its own [`SteppableSolver`].
129///
130/// See [module documentation](self) for the per-step order and the v1
131/// synchronised-`dt` scope.
132#[derive(Default)]
133pub struct MultiDomainOrchestrator {
134    solvers: HashMap<DomainId, Box<dyn SteppableSolver>>,
135    /// The `PhysicalQuantity` each domain's primary state is keyed under
136    /// in `MultiDomainState` — convention established by
137    /// `tests/coupling_proto.rs` (v0.3.0): the caller picks one explicitly,
138    /// `PhysicalModel` does not declare it itself.
139    quantities: HashMap<DomainId, PhysicalQuantity>,
140}
141
142impl MultiDomainOrchestrator {
143    /// Creates an empty orchestrator with no domains registered.
144    pub fn new() -> Self {
145        Self {
146            solvers: HashMap::new(),
147            quantities: HashMap::new(),
148        }
149    }
150
151    /// Registers the solver and state quantity key to use for `domain_id`.
152    ///
153    /// Every domain present in the `Scenario` passed to [`run`](Self::run)
154    /// must have a corresponding entry, or `run` returns
155    /// `OxiflowError::InvalidDomain`.
156    ///
157    /// # Examples
158    ///
159    /// ```rust,ignore
160    /// let orchestrator = MultiDomainOrchestrator::new()
161    ///     .with_domain(lahar_id, Box::new(ForwardEulerSolver), PhysicalQuantity::concentration())
162    ///     .with_domain(lake_id, Box::new(RK4Solver), PhysicalQuantity::concentration());
163    /// ```
164    pub fn with_domain(
165        mut self,
166        domain_id: DomainId,
167        solver: Box<dyn SteppableSolver>,
168        quantity: PhysicalQuantity,
169    ) -> Self {
170        self.solvers.insert(domain_id.clone(), solver);
171        self.quantities.insert(domain_id, quantity);
172        self
173    }
174
175    /// Runs the orchestrated simulation and returns the collected states.
176    ///
177    /// # Errors
178    ///
179    /// - `OxiflowError::InvalidDomain` if any domain in `scenario` has no
180    ///   registered solver/quantity, if `scenario` has no domains, or if
181    ///   `dt`/`t_end` are invalid.
182    /// - `OxiflowError::InvalidDomain` if `config.time.step_control` is not
183    ///   `StepControl::Fixed` (DD-031 v1 scope).
184    pub fn run(
185        &self,
186        scenario: &Scenario,
187        config: &OrchestratorConfig,
188    ) -> Result<MultiDomainSimulationResult, OxiflowError> {
189        scenario.validate()?;
190
191        if scenario.n_domains() == 0 {
192            return Err(OxiflowError::InvalidDomain(
193                "scenario has no domains".into(),
194            ));
195        }
196        for domain in scenario.domains() {
197            if !self.solvers.contains_key(&domain.id) {
198                return Err(OxiflowError::InvalidDomain(format!(
199                    "no SteppableSolver registered for domain '{}' -- call with_domain() for it",
200                    domain.id
201                )));
202            }
203        }
204
205        let dt = match &config.time.step_control {
206            StepControl::Fixed { dt } => *dt,
207            _ => {
208                return Err(OxiflowError::InvalidDomain(
209                    "MultiDomainOrchestrator only supports StepControl::Fixed \
210                     (DD-031 v1: dt synchronised across all domains)"
211                        .into(),
212                ))
213            }
214        };
215
216        let t_end = config.time.t_end;
217        let t_start = scenario.t_start;
218
219        if dt <= 0.0 {
220            return Err(OxiflowError::InvalidDomain(
221                "dt must be strictly positive".into(),
222            ));
223        }
224        if t_end <= t_start {
225            return Err(OxiflowError::InvalidDomain(
226                "t_end must be greater than t_start".into(),
227            ));
228        }
229
230        // One shared calculator chain, built from the aggregated
231        // requirements of every domain + coupling — same convention
232        // `Solver::solve()` uses for the single-domain case.
233        let requirements = scenario.context_requirements();
234        let chain = build_calculator_chain(&requirements, &config.calculators)?;
235
236        let n_steps = ((t_end - t_start) / dt).round() as usize;
237        let save_every = config.time.save_every.unwrap_or(1);
238        let capacity = n_steps / save_every + 1;
239
240        // ── Initial state — one entry per domain ────────────────────────────
241        let mut multi_state = MultiDomainState::new();
242        for domain in scenario.domains() {
243            let quantity = self.quantity_for(&domain.id)?;
244            let initial = domain.model.initial_state(domain.mesh.as_ref());
245            multi_state.set(domain.id.clone(), quantity, initial);
246        }
247
248        // Per-domain history buffers, sized to each domain's own solver
249        // (DD-034) — empty for one-step methods (history_depth() == 0,
250        // the default), populated for multi-step ones (BDF2: depth 1).
251        let mut histories: HashMap<DomainId, Vec<ContextValue>> = scenario
252            .domains()
253            .iter()
254            .map(|d| (d.id.clone(), Vec::new()))
255            .collect();
256
257        let mut states: Vec<MultiDomainState> = Vec::with_capacity(capacity);
258        let mut times: Vec<f64> = Vec::with_capacity(capacity);
259        states.push(multi_state.clone());
260        times.push(t_start);
261
262        // ── Time loop ────────────────────────────────────────────────────────
263        for step in 0..n_steps {
264            let t = t_start + (step as f64) * dt;
265            let t_next = t_start + ((step + 1) as f64) * dt;
266
267            // 1. Advance every domain by one step, each with its own solver.
268            for domain in scenario.domains() {
269                let quantity = self.quantity_for(&domain.id)?;
270                let solver = &self.solvers[&domain.id];
271
272                let mut state = multi_state
273                    .get(&domain.id, &quantity)
274                    .ok_or_else(|| {
275                        OxiflowError::InvalidDomain(format!(
276                            "domain '{}' has no state for its registered quantity",
277                            domain.id
278                        ))
279                    })?
280                    .clone();
281
282                let next_state = {
283                    let history = &histories[&domain.id];
284                    solver.step(domain, &chain, &mut state, history, t, dt)?
285                };
286
287                // `state` was mutated in-place by BC application inside
288                // `step()` above -- push *that* corrected u^n into history
289                // (depth-capped per the solver's own declared need), not
290                // the pre-correction value.
291                let depth = solver.history_depth();
292                if depth > 0 {
293                    let hist = histories.get_mut(&domain.id).unwrap();
294                    hist.insert(0, state);
295                    hist.truncate(depth);
296                }
297
298                multi_state.set(domain.id.clone(), quantity, next_state);
299            }
300
301            // 2. Exchange state across every coupling interface, in
302            //    declaration order -- each operator sees the result of the
303            //    previous one.
304            let ctx = ComputeContext::new(t_next, dt);
305            for coupling in scenario.couplings() {
306                multi_state = coupling.apply(&multi_state, &ctx, coupling.interface())?;
307            }
308
309            // 3. Guard against divergence across all domains.
310            for domain in scenario.domains() {
311                let quantity = self.quantity_for(&domain.id)?;
312                if let Some(value) = multi_state.get(&domain.id, &quantity) {
313                    check_finite(value, t_next)?;
314                }
315            }
316
317            if (step + 1) % save_every == 0 {
318                states.push(multi_state.clone());
319                times.push(t_next);
320            }
321        }
322
323        Ok(MultiDomainSimulationResult {
324            states,
325            times,
326            n_steps,
327        })
328    }
329
330    /// Looks up the registered quantity for `domain_id`.
331    fn quantity_for(&self, domain_id: &DomainId) -> Result<PhysicalQuantity, OxiflowError> {
332        self.quantities.get(domain_id).cloned().ok_or_else(|| {
333            OxiflowError::InvalidDomain(format!(
334                "no PhysicalQuantity key registered for domain '{domain_id}' -- call with_domain() for it"
335            ))
336        })
337    }
338}

416    /// Transfers a fixed fraction of the source domain's field to the
417    /// target domain at every call -- counts invocations for assertions.
418    #[derive(Debug)]
419    struct CountingMassTransfer {
420        alpha: f64,
421        interface: Interface,
422        calls: std::sync::Arc<std::sync::atomic::AtomicUsize>,
423    }
424
425    impl RequiresContext for CountingMassTransfer {
426        fn required_variables(&self) -> Vec<ContextVariable> {
427            vec![]
428        }
429    }
430
431    impl CouplingOperator for CountingMassTransfer {
432        fn apply(
433            &self,
434            states: &MultiDomainState,
435            _ctx: &ComputeContext,
436            interface: &Interface,
437        ) -> Result<MultiDomainState, OxiflowError> {
438            self.calls.fetch_add(1, std::sync::atomic::Ordering::SeqCst);
439
440            let quantity = PhysicalQuantity::concentration();
441            let source = states
442                .get(interface.source(), &quantity)
443                .ok_or_else(|| OxiflowError::InvalidDomain("missing source field".into()))?
444                .as_scalar_field()?;
445            let transferred = source.map(|v| self.alpha * v);
446
447            let mut result = states.clone();
448            result.set(
449                interface.target().clone(),
450                quantity,
451                ContextValue::ScalarField(transferred),
452            );
453            Ok(result)
454        }
455
456        fn interface(&self) -> &Interface {
457            &self.interface
458        }
459    }
460
461    fn source_id() -> DomainId {
462        DomainId::new("source")
463    }
464    fn target_id() -> DomainId {
465        DomainId::new("target")

468    fn make_mesh(n: usize) -> Box<dyn Mesh> {
469        Box::new(UniformGrid1D::new(n, 0.0, 1.0).unwrap())

472    fn make_scenario() -> (Scenario, std::sync::Arc<std::sync::atomic::AtomicUsize>) {
473        let source = Domain::new(
474            source_id(),
475            Box::new(DecayModel { lambda: 0.5 }),
476            make_mesh(3),
477        );
478        let target = Domain::new(target_id(), Box::new(PassiveModel), make_mesh(3));
479        let interface = Interface::new(source_id(), target_id());
480        let calls = std::sync::Arc::new(std::sync::atomic::AtomicUsize::new(0));
481        let coupling = Box::new(CountingMassTransfer {
482            alpha: 0.1,
483            interface,
484            calls: calls.clone(),
485        });
486        let scenario = Scenario::multi(vec![source, target])
487            .unwrap()
488            .with_coupling(coupling);
489        (scenario, calls)
490    }

566    #[test]
567    fn coupling_invoked_exactly_once_per_step() {
568        let (scenario, calls) = make_scenario();
569        let orchestrator = MultiDomainOrchestrator::new()
570            .with_domain(
571                source_id(),
572                Box::new(ForwardEulerSolver),
573                PhysicalQuantity::concentration(),
574            )
575            .with_domain(
576                target_id(),
577                Box::new(ForwardEulerSolver),
578                PhysicalQuantity::concentration(),
579            );
580
581        let result = orchestrator.run(&scenario, &make_config(0.5, 0.1)).unwrap();
582        assert_eq!(result.n_steps, 5);
583
584        // 5 steps -> exactly 5 invocations, not 0 (never called) and not
585        // more (e.g. accidentally invoked once per domain per step).
586        assert_eq!(calls.load(std::sync::atomic::Ordering::SeqCst), 5);
587
588        // Cross-check via effect: target received non-zero mass, which
589        // only the coupling (never PassiveModel itself, zero derivative)
590        // can produce.
591        let final_state = result.states.last().unwrap();
592        let target_field = final_state
593            .get(&target_id(), &PhysicalQuantity::concentration())
594            .unwrap()
595            .as_scalar_field()
596            .unwrap();
597        assert!(target_field.iter().all(|v| *v > 0.0));
598    }
