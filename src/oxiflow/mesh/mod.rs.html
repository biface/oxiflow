3//! Spatial mesh abstraction — INV-1 invariant (DD-007, DD-019, issue #31).
4//!
5//! ## Core principle (INV-1)
6//!
7//! No public engine API exposes `dx`, `nx` or raw indices as first-class spatial
8//! parameters. All spatial references go through the `Mesh` trait. This guarantees
9//! forward compatibility with FEM unstructured meshes at J7 — zero breaking change
10//! on existing code when `unstructured/` is added.

41/// Abstract spatial mesh — INV-1 invariant.
42///
43/// All spatial information consumed by the engine passes through this trait.
44/// No implementation detail (`dx`, `nx`, node table) leaks into the public API.
45///
46/// # Object safety
47///
48/// This trait is object-safe: it can be used as `Box<dyn Mesh>` and `&dyn Mesh`.
49/// Required for INV-4 (plugin-safe API, J7).
50///
51/// # Implementing `Mesh`
52///
53/// A minimal implementation for a 1D uniform grid:
54///
55/// ```rust
56/// use oxiflow::mesh::Mesh;
57///
58/// struct MyGrid { n: usize, dx: f64 }
59///
60/// impl Mesh for MyGrid {
61///     fn n_dof(&self) -> usize { self.n }
62///     fn coordinates(&self, i: usize) -> &[f64] {
63///         // In practice, return a slice into a pre-computed node table.
64///         // This example uses a workaround for illustration only.
65///         std::slice::from_ref(Box::leak(Box::new(i as f64 * self.dx)))
66///     }
67///     fn spatial_dimension(&self) -> usize { 1 }
68///     fn characteristic_length(&self) -> f64 { self.dx }
69/// }
70/// ```
71///
72/// # INV-1 compliance
73///
74/// Implementations must **not** expose `dx`, `nx` or raw indices in any public
75/// method beyond those defined here. Spatial parameters are internal details.
76pub trait Mesh: Send + Sync {
77    /// Total number of degrees of freedom (nodes) in the mesh.
78    ///
79    /// For a 1D uniform grid: `n_dof() == n_points`.
80    /// For a 2D FEM mesh: `n_dof()` is the number of mesh nodes.
81    fn n_dof(&self) -> usize;
82
83    /// Coordinates of node `i` as a slice of length `spatial_dimension()`.
84    ///
85    /// Returns `&[f64]` — zero allocation. Callers must not store the returned
86    /// reference beyond the lifetime of the mesh. Implementors typically return
87    /// a slice into a pre-computed node coordinate table.
88    ///
89    /// # Panics
90    ///
91    /// Implementations may panic if `i >= n_dof()`. Use `n_dof()` to guard
92    /// iteration bounds.
93    fn coordinates(&self, i: usize) -> &[f64];
94
95    /// Number of spatial dimensions of this mesh.
96    ///
97    /// - `1` for 1D grids (chromatography column, 1D heat transfer)
98    /// - `2` for 2D meshes (surface flow, 2D diffusion)
99    /// - `3` for 3D meshes (volumetric FEM)
100    fn spatial_dimension(&self) -> usize;
101
102    /// A representative spatial length scale of the mesh.
103    ///
104    /// Used by integrators and operators for stability estimates (CFL condition,
105    /// Péclet number). For a uniform 1D grid: `dx`. For unstructured meshes:
106    /// the minimum element diameter or average edge length.
107    fn characteristic_length(&self) -> f64;
108}
